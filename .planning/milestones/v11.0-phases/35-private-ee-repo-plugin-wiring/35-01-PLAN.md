    ```python
    from __future__ import annotations
    import logging

    logger = logging.getLogger(__name__)


    class EEPlugin:
        """Entry-point class discovered by CE via importlib.metadata.
        Entry point: axiom.ee = ee.plugin:EEPlugin

        IMPORTANT: No module-level router imports here. All router imports
        are deferred inside register() to prevent circular imports when
        CE loads this plugin before agent_service.main finishes startup.
        """

        def __init__(self, app, engine) -> None:
            self._app = app
            self._engine = engine

        async def register(self, ctx) -> None:
            """Create EE tables and mount EE routers.

            Order is deliberate:
            1. Create all EE tables (sync DDL via sync_engine)
            2. Mount each router, set feature flag only on success
            3. Set resource_limits flag unconditionally (DB-level capability)
            """
            from ee.base import EEBase
            # Step 1: Create all EE tables before any router goes live.
            # EEBase.metadata.create_all is synchronous — use sync_engine,
            # never pass the AsyncEngine directly (raises TypeError in SA 2.x).
            EEBase.metadata.create_all(self._engine.sync_engine)
            logger.info("EE tables created via EEBase.metadata.create_all")

            # Step 2-3: Router mounting — implemented in Plan 03.
            # Stub placeholders so the package is importable now:
            # (replace these comments with real includes in plan 03)
            ctx.resource_limits = True
            logger.info("EEPlugin.register() complete (stub — routers added in plan 03)")
    ```
