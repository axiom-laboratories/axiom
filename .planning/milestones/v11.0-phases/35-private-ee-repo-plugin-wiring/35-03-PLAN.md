    ```python
    async def register(self, ctx) -> None:
        from ee.base import EEBase
        # 1. Import all EE model modules to register them with EEBase.metadata
        import ee.foundry.models
        import ee.smelter.models
        import ee.audit.models
        import ee.auth_ext.models
        import ee.rbac.models
        import ee.webhooks.models
        import ee.triggers.models

        # 2. Create all EE tables (sync DDL — must use sync_engine, not AsyncEngine)
        EEBase.metadata.create_all(self._engine.sync_engine)
        logger.info("EE tables created via EEBase.metadata.create_all")

        # 3. Mount routers — deferred imports to prevent circular import at module load time
        try:
            from ee.foundry.router import foundry_router
            self._app.include_router(foundry_router)
            ctx.foundry = True
            logger.info("EE foundry router mounted")
        except Exception as e:
            logger.warning(f"EE foundry router failed to mount: {e}")

        try:
            from ee.smelter.router import smelter_router
            self._app.include_router(smelter_router)
            # smelter is part of foundry feature — no separate flag
            logger.info("EE smelter router mounted")
        except Exception as e:
            logger.warning(f"EE smelter router failed to mount: {e}")

        try:
            from ee.audit.router import audit_router
            self._app.include_router(audit_router)
            ctx.audit = True
            logger.info("EE audit router mounted")
        except Exception as e:
            logger.warning(f"EE audit router failed to mount: {e}")

        try:
            from ee.webhooks.router import webhook_router
            self._app.include_router(webhook_router)
            ctx.webhooks = True
            logger.info("EE webhooks router mounted")
        except Exception as e:
            logger.warning(f"EE webhooks router failed to mount: {e}")

        try:
            from ee.triggers.router import trigger_router
            self._app.include_router(trigger_router)
            ctx.triggers = True
            logger.info("EE triggers router mounted")
        except Exception as e:
            logger.warning(f"EE triggers router failed to mount: {e}")

        try:
            from ee.users.router import users_router
            self._app.include_router(users_router)
            ctx.rbac = True
            logger.info("EE users/RBAC router mounted")
        except Exception as e:
            logger.warning(f"EE users router failed to mount: {e}")

            from ee.auth_ext.router import auth_ext_router
            self._app.include_router(auth_ext_router)

        # resource_limits is a DB-level capability (no dedicated router)
        ctx.resource_limits = True
        logger.info(f"EEPlugin.register() complete. Flags: {ctx}")
    ```
