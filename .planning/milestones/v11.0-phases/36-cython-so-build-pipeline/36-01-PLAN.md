```python
# setup.py — Cython compilation config for axiom-ee
# Uses packages=[] to exclude .py source from wheel; .so files included via ext_modules.
# BuildExtAndCopyInits copies __init__.py namespace markers into build dir after compilation.
# Reference: https://bucharjan.cz/blog/using-cython-to-protect-a-python-codebase.html
import glob
import shutil
from pathlib import Path
from setuptools import setup
from setuptools.extension import Extension
from Cython.Build import cythonize
from setuptools.command.build_ext import build_ext as _build_ext


class BuildExtAndCopyInits(_build_ext):
    """Copy __init__.py files into build dir after .so compilation.
    Required because packages=[] excludes them from normal package collection."""

    def run(self):
        _build_ext.run(self)
        build_dir = Path(self.build_lib)
        root_dir = Path(__file__).parent
        target_dir = build_dir if not self.inplace else root_dir
        for init_file in root_dir.glob("ee/**/__init__.py"):
            rel = init_file.relative_to(root_dir)
            dest = target_dir / rel
            dest.parent.mkdir(parents=True, exist_ok=True)
            shutil.copyfile(str(init_file), str(dest))


# Collect all .py files except __init__.py — these become .so extensions
SOURCES = sorted(
    f for f in glob.glob("ee/**/*.py", recursive=True)
    if not f.endswith("__init__.py")
)

ext_modules = [
    Extension(
        name=src.replace("/", ".").removesuffix(".py"),
        sources=[src],
    )
    for src in SOURCES
]

setup(
    ext_modules=cythonize(
        ext_modules,
        compiler_directives={
            "language_level": "3",
        },
        nthreads=4,
    ),
    cmdclass={"build_ext": BuildExtAndCopyInits},
    packages=[],  # CRITICAL: strips .py source from wheel; .so included via ext_modules
)
```

Verify SOURCES will contain exactly the 21 files listed in context. The `language_level=3` directive is required for `from __future__ import annotations` in all EE source files (Cython 3.2.4 handles this correctly).

Do NOT compile `__init__.py` files — the glob filter `if not f.endswith("__init__.py")` handles this. The `BuildExtAndCopyInits` hook copies them back as plain Python namespace markers.

**Create `~/Development/axiom-ee/Makefile`:**

```makefile
.PHONY: build clean upload qemu-setup

# Register QEMU for arm64 emulation (idempotent — safe to run multiple times)
qemu-setup:
	docker run --rm --privileged tonistiigi/binfmt --install all

# Build multi-arch, multi-Python wheels using cibuildwheel
# Output: wheelhouse/ directory (git-ignored)
build: qemu-setup
	cibuildwheel --platform linux .

# Upload wheels to local devpi server
# Requires: DEVPI_PASSWORD env var (default: changeme)
upload:
	twine upload \
	  --repository-url http://localhost:3141/root/dev/ \
	  --non-interactive \
	  -u root -p $${DEVPI_PASSWORD:-changeme} \
	  wheelhouse/*.whl

clean:
	rm -rf wheelhouse/ build/ *.egg-info ee/**/*.c ee/**/*.so
```
