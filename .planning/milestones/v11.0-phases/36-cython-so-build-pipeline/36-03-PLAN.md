```python
#!/usr/bin/env python3
"""
Phase 36 — BUILD-05: Compiled wheel CE+EE smoke test.

Validates that axiom-ee 0.1.0 compiled .so wheel produces identical CE+EE
behaviour to the Phase 35 Python source install.

Usage:
    python mop_validation/scripts/test_compiled_wheel.py

Pre-conditions:
    1. devpi running on localhost:3141 with axiom-ee wheels uploaded
    2. .worktrees/axiom-split/puppeteer/Containerfile.server has EE_INSTALL ARG
    3. wheelhouse/ contains compiled wheels
    4. mop_validation/secrets.env has ADMIN_PASSWORD, AGENT_URL, API_KEY

Tests:
    - GET /api/features returns all 8 EE flags True
    - GET /api/blueprints returns 200 (not 402)
    - GET /api/features in CE-only mode returns all flags False (regression check)
"""
import subprocess
import sys
import time
import os
from pathlib import Path

import requests

WORKTREE = Path.home() / "Development/master_of_puppets/.worktrees/axiom-split/puppeteer"
SECRETS = Path.home() / "Development/mop_validation/secrets.env"
AGENT_URL = "https://localhost:8001"


def load_secrets():
    """Load secrets.env into environment."""
    if not SECRETS.exists():
        print(f"ERROR: {SECRETS} not found")
        sys.exit(1)
    with open(SECRETS) as f:
        for line in f:
            line = line.strip()
            if line and not line.startswith("#") and "=" in line:
                k, _, v = line.partition("=")
                os.environ.setdefault(k.strip(), v.strip())


def run(cmd, cwd=None, check=True, capture=False):
    """Run a shell command."""
    kwargs = {"shell": True, "cwd": cwd or WORKTREE}
    if capture:
        kwargs["capture_output"] = True
        kwargs["text"] = True
    print(f"  $ {cmd}")
    result = subprocess.run(cmd, **kwargs)
    if check and result.returncode != 0:
        print(f"FAIL: command exited {result.returncode}")
        sys.exit(1)
    return result


def wait_for_api(url, timeout=60, token=None):
    """Poll API until it responds 200 or raises on timeout."""
    headers = {}
    if token:
        headers["Authorization"] = f"Bearer {token}"
    deadline = time.time() + timeout
    while time.time() < deadline:
        try:
            r = requests.get(url, headers=headers, verify=False, timeout=5)
            if r.status_code < 500:
                return r
        except Exception:
            pass
        time.sleep(2)
    print(f"TIMEOUT: {url} did not respond within {timeout}s")
    sys.exit(1)


def get_token():
    """Authenticate and return JWT."""
    password = os.environ.get("ADMIN_PASSWORD", "admin")
    r = requests.post(
        f"{AGENT_URL}/api/auth/login",
        json={"username": "admin", "password": password},
        verify=False,
        timeout=10,
    )
    assert r.status_code == 200, f"Login failed: {r.status_code} {r.text}"
    return r.json()["access_token"]


def test_ce_only_features_false():
    """CE-only stack: all features must be False."""
    print("\n[CE-only] Bringing up CE stack (no EE_INSTALL)...")
    run("docker compose -f compose.server.yaml down -v --remove-orphans 2>/dev/null || true")
    run("docker compose -f compose.server.yaml build agent")
    run("docker compose -f compose.server.yaml up -d db agent")
    time.sleep(15)
    wait_for_api(f"{AGENT_URL}/api/features")
    token = get_token()
    r = requests.get(
        f"{AGENT_URL}/api/features",
        headers={"Authorization": f"Bearer {token}"},
        verify=False,
    )
    assert r.status_code == 200, f"features returned {r.status_code}"
    flags = r.json()
    false_flags = [k for k, v in flags.items() if v is not False]
    assert not false_flags, f"CE mode: expected all False, got True for: {false_flags}"
    print(f"  PASS: CE mode — all {len(flags)} features False")


def test_ee_compiled_features_true():
    """CE+EE compiled wheel: all features must be True, /api/blueprints must return 200."""
    print("\n[EE compiled] Bringing up CE+EE stack with compiled wheel...")
    run("docker compose -f compose.server.yaml down -v --remove-orphans 2>/dev/null || true")
    run("docker compose -f compose.server.yaml build --build-arg EE_INSTALL=1 agent")
    run("docker compose -f compose.server.yaml up -d db agent devpi")
    time.sleep(20)
    wait_for_api(f"{AGENT_URL}/api/features")
    token = get_token()
    headers = {"Authorization": f"Bearer {token}"}

    # Check features
    r = requests.get(f"{AGENT_URL}/api/features", headers=headers, verify=False)
    assert r.status_code == 200
    flags = r.json()
    false_flags = [k for k, v in flags.items() if v is not True]
    assert not false_flags, f"EE mode: expected all True, got False for: {false_flags}"
    print(f"  PASS: EE compiled mode — all {len(flags)} features True")

    # Check EE route is live (not 402)
    r2 = requests.get(f"{AGENT_URL}/api/blueprints", headers=headers, verify=False)
    assert r2.status_code != 402, f"/api/blueprints returned 402 — EE router not active"
    assert r2.status_code == 200, f"/api/blueprints returned {r2.status_code}"
    print(f"  PASS: /api/blueprints returned {r2.status_code} (EE router live)")


def main():
    import urllib3
    urllib3.disable_warnings()

    print("=== Phase 36 BUILD-05: Compiled Wheel CE+EE Smoke Test ===")
    load_secrets()

    try:
        test_ce_only_features_false()
        test_ee_compiled_features_true()
        print("\n=== ALL TESTS PASSED — BUILD-05 COMPLETE ===")
    finally:
        # Leave stack running for inspection; caller can run docker compose down
        print("\nStack left running. To clean up:")
        print(f"  cd {WORKTREE} && docker compose -f compose.server.yaml down -v")


if __name__ == "__main__":
    main()
```
