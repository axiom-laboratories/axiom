```python
@app.patch("/nodes/{node_id}/drain", tags=["Nodes"])
async def drain_node(node_id: str, current_user: User = Depends(require_permission("nodes:write")), db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(Node).where(Node.node_id == node_id))
    node = result.scalar_one_or_none()
    if not node:
        raise HTTPException(status_code=404, detail="Node not found")
    if node.status not in ("ONLINE", "BUSY"):
        raise HTTPException(status_code=409, detail=f"Cannot drain node in {node.status} state")
    node.status = "DRAINING"
    audit(db, current_user, "node:drain", node_id)
    await db.commit()
    await ws_manager.broadcast("node:updated", {"node_id": node_id, "status": "DRAINING"})
    return {"status": "DRAINING", "node_id": node_id}

@app.patch("/nodes/{node_id}/undrain", tags=["Nodes"])
async def undrain_node(node_id: str, current_user: User = Depends(require_permission("nodes:write")), db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(Node).where(Node.node_id == node_id))
    node = result.scalar_one_or_none()
    if not node:
        raise HTTPException(status_code=404, detail="Node not found")
    if node.status != "DRAINING":
        raise HTTPException(status_code=409, detail="Node is not in DRAINING state")
    node.status = "ONLINE"
    audit(db, current_user, "node:undrain", node_id)
    await db.commit()
    await ws_manager.broadcast("node:updated", {"node_id": node_id, "status": "ONLINE"})
    return {"status": "ONLINE", "node_id": node_id}
```

**Step 1: job_service.py — get_dispatch_diagnosis static async method**

Add `get_dispatch_diagnosis(guid: str, db: AsyncSession) -> dict` as a static async method on `JobService`. Logic:

```python
@staticmethod
async def get_dispatch_diagnosis(guid: str, db: AsyncSession) -> dict:
    """Returns {reason, message, queue_position} for a PENDING job."""
    # 1. Load job
    job_result = await db.execute(select(Job).where(Job.guid == guid))
    job = job_result.scalar_one_or_none()
    if not job:
        return {"reason": "not_found", "message": "Job not found", "queue_position": None}
    if job.status != "PENDING":
        return {"reason": "not_pending", "message": f"Job is {job.status}, not PENDING", "queue_position": None}

    # 2. Handle explicit target_node_id targeting
    # (target_tags may contain a node_id prefixed value — check if single exact node targeted)
    # Check if any ONLINE/BUSY node exists at all
    nodes_result = await db.execute(
        select(Node).where(Node.status.in_(["ONLINE", "BUSY"]))
    )
    online_nodes = nodes_result.scalars().all()

    if not online_nodes:
        return {
            "reason": "no_nodes_online",
            "message": "No nodes are currently online. The job will dispatch when a node connects.",
            "queue_position": None
        }

    # 3. Find eligible nodes using _node_is_eligible helper
    eligible_nodes = []
    missing_cap = None
    for node in online_nodes:
        node_tags = JobService._get_effective_tags(node)
        node_caps_dict = json.loads(node.capabilities) if node.capabilities else {}
        # Check target_node_id first (encoded as a tag or direct field)
        if JobService._node_is_eligible(node, job, node_tags, node_caps_dict):
            eligible_nodes.append(node)
        else:
            # Try to identify the missing capability for better error messages
            if job.capability_requirements:
                try:
                    req_caps = json.loads(job.capability_requirements)
                    for cap_name, min_ver in req_caps.items():
                        if cap_name not in node_caps_dict:
                            missing_cap = cap_name
                            break
                except Exception:
                    pass

    if not eligible_nodes:
        msg = f"No nodes match this job's requirements."
        if missing_cap:
            msg = f"No nodes have the required capability '{missing_cap}'."
        return {"reason": "capability_mismatch", "message": msg, "queue_position": None}

    # 4. Check if all eligible nodes are at concurrency limit
    all_busy = True
    for node in eligible_nodes:
        assigned_result = await db.execute(
            select(func.count(Job.guid)).where(
                Job.status == 'ASSIGNED',
                Job.node_id == node.node_id
            )
        )
        assigned = assigned_result.scalar() or 0
        limit = node.concurrency_limit or 5
        if assigned < limit:
            all_busy = False
            break

    if all_busy:
        # Count jobs ahead in queue (created before this job, PENDING/RETRYING)
        pos_result = await db.execute(
            select(func.count(Job.guid)).where(
                Job.status.in_(["PENDING", "RETRYING"]),
                Job.created_at < job.created_at
            )
        )
        queue_position = (pos_result.scalar() or 0) + 1
        return {
            "reason": "all_nodes_busy",
            "message": f"All eligible nodes are at capacity. Approximately {queue_position - 1} jobs ahead in queue.",
            "queue_position": queue_position
        }

    return {
        "reason": "pending_dispatch",
        "message": "This job will be dispatched on the next poll cycle.",
        "queue_position": 1
    }
```

Also handle the target_node_offline case: if `job.target_tags` contains exactly one entry that starts with "node:" and that node is OFFLINE or DRAINING, return reason="target_node_unavailable".
