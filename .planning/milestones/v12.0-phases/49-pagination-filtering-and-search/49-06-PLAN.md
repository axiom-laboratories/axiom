```typescript
const [page, setPage] = useState(1);
```

<!-- Updated fetchNodes -->
```typescript
const fetchNodes = async (p = page) => {
    const res = await authenticatedFetch(`/nodes?page=${p}&page_size=${PAGE_SIZE}`);
    if (!res.ok) return;
    const data: PaginatedNodeResponse = await res.json();
    setNodes(data.items);
    setTotalNodes(data.total);
    setTotalPages(data.pages);
};
```

<!-- Page controls JSX (add below the nodes grid/table) -->
```tsx
<div className="flex items-center justify-between mt-4">
    <span className="text-sm text-muted-foreground">
        Showing {nodes.length} of {totalNodes} nodes
    </span>
    <div className="flex items-center gap-2">
        <Button
            variant="outline"
            size="sm"
            disabled={page <= 1}
            onClick={() => { setPage(p => p - 1); fetchNodes(page - 1); }}
        >
            Previous
        </Button>
        <span className="text-sm">Page {page} of {totalPages}</span>
        <Button
            variant="outline"
            size="sm"
            disabled={page >= totalPages}
            onClick={() => { setPage(p => p + 1); fetchNodes(page + 1); }}
        >
            Next
        </Button>
    </div>
</div>
```
