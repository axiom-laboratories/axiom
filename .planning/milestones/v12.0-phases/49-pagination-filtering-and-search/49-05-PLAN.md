```typescript
const fetchJobs = async (opts: { reset?: boolean; cursor?: string | null } = {}) => {
    const params = new URLSearchParams();
    if (opts.cursor) params.set('cursor', opts.cursor);
    if (filters.status !== 'all') params.set('status', filters.status);
    if (filters.runtime !== 'all') params.set('runtime', filters.runtime);
    if (filters.taskType !== 'all') params.set('task_type', filters.taskType);
    if (filters.nodeId) params.set('node_id', filters.nodeId);
    if (filters.tags.length) params.set('tags', filters.tags.join(','));
    if (filters.createdBy) params.set('created_by', filters.createdBy);
    if (filters.dateFrom) params.set('date_from', filters.dateFrom);
    if (filters.dateTo) params.set('date_to', filters.dateTo);
    if (filters.search) params.set('search', filters.search);

    const res = await authenticatedFetch(`/jobs?${params}`);
    if (!res.ok) return;
    const data: PaginatedJobResponse = await res.json();

    if (opts.reset) {
        setJobs(data.items);
    } else {
        setJobs(prev => [...prev, ...data.items]);  // "load more" appends
    }
    setTotal(data.total);
    setNextCursor(data.next_cursor);
    setPendingNewJobs(0);
};
```

useWebSocket((event, data) => {
    if (event === 'job:created') {
        setPendingNewJobs(c => c + 1);   // increment banner; do NOT refetch
    } else if (event === 'job:updated') {
        setJobs(prev => prev.map(j => j.guid === data.guid ? { ...j, ...data } : j));
    }
});

<!-- Export handler -->
```typescript
const handleExport = async () => {
    const params = new URLSearchParams();
    // same filter params as fetchJobs, without cursor
    const res = await authenticatedFetch(`/jobs/export?${params}`);
    if (!res.ok) return;
    const blob = await res.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'jobs-export.csv';
    a.click();
    URL.revokeObjectURL(url);
};
```
