    ```python
    #!/usr/bin/env python3
    """
    run_ee_scenario.py — EE cold-start orchestrator (Phase 64).

    Extends run_ce_scenario.py with EE-specific reset (licence key injection),
    node enrollment polling, CE-gating confirmation, and EE FRICTION file pull.

    Exported functions:
      read_ee_licence_key() -> str
      reset_stack_ee() -> None
      wait_for_node_enrollment(admin_token, timeout) -> bool
      pull_ee_friction(scenario_id, reports_dir) -> bool
      confirm_ce_gating() -> bool

    Standalone usage (rebuild image, reset EE stack, verify readiness):
      python3 run_ee_scenario.py
    """

    import subprocess
    import sys
    import time
    from pathlib import Path

    # Import proven CE helpers — do not re-implement
    sys.path.insert(0, str(Path(__file__).parent))
    from run_ce_scenario import incus_exec, incus_push, incus_pull, wait_for_stack, CONTAINER

    SECRETS_ENV = "/home/thomas/Development/mop_validation/secrets.env"
    COMPOSE_SRC = "/home/thomas/Development/master_of_puppets/puppeteer/compose.cold-start.yaml"
    REPORTS_DIR = "/home/thomas/Development/mop_validation/reports"
    SERVER_IMAGE = "localhost/master-of-puppets-server:v3"
    PUPPETEER_DIR = "/home/thomas/Development/master_of_puppets/puppeteer"

    # ---------------------------------------------------------------------------
    # EE licence key
    # ---------------------------------------------------------------------------

    def read_ee_licence_key() -> str:
        """Read AXIOM_EE_LICENCE_KEY from host secrets.env."""
        with open(SECRETS_ENV) as f:
            for line in f:
                if line.startswith("AXIOM_EE_LICENCE_KEY="):
                    return line.strip().split("=", 1)[1]
        raise RuntimeError("AXIOM_EE_LICENCE_KEY not found in secrets.env")

    # ---------------------------------------------------------------------------
    # EE image build
    # ---------------------------------------------------------------------------

    def build_ee_server_image() -> None:
        """Build server image with EE_INSTALL=1 (uses local wheel COPY)."""
        print("Building EE server image (EE_INSTALL=1)...")
        subprocess.run(
            [
                "docker", "compose",
                "-f", f"{PUPPETEER_DIR}/compose.cold-start.yaml",
                "build",
                "--build-arg", "EE_INSTALL=1",
                "agent",
            ],
            check=True,
            cwd=PUPPETEER_DIR,
        )
        print("  Build complete.")

    def push_ee_image_to_lxc() -> None:
        """Save EE server image on host, load into LXC via docker save | docker load."""
        print(f"Pushing {SERVER_IMAGE} into LXC...")
        save = subprocess.Popen(
            ["docker", "save", SERVER_IMAGE],
            stdout=subprocess.PIPE,
        )
        load = subprocess.Popen(
            ["incus", "exec", CONTAINER, "--", "docker", "load"],
            stdin=save.stdout,
        )
        save.stdout.close()
        load.communicate()
        if load.returncode != 0:
            raise RuntimeError("docker load inside LXC failed")
        print("  Image loaded in LXC.")

    def smoke_check_ee_image() -> None:
        """Verify ee.plugin is importable in the server image inside the LXC."""
        print("Smoke-checking EE plugin import in LXC...")
        result = incus_exec(
            f"docker run --rm {SERVER_IMAGE} "
            "python -c \"import ee.plugin; print('EE OK')\"",
            timeout=60,
        )
        if "EE OK" not in result.stdout:
            print(f"  STDOUT: {result.stdout}")
            print(f"  STDERR: {result.stderr}")
            raise RuntimeError("EE smoke check FAILED: ee.plugin not importable in server image")
        print("  EE smoke check PASSED.")

    # ---------------------------------------------------------------------------
    # EE stack reset
    # ---------------------------------------------------------------------------

    def reset_stack_ee() -> None:
        """
        Full cold-start reset with EE licence key injected.

        - Tears down with -v (wipes certs, state)
        - Writes AXIOM_LICENCE_KEY to /workspace/.env
        - Starts with --env-file .env to ensure key reaches agent container
        - Cleans stale EE friction/checkpoint files
        """
        licence_key = read_ee_licence_key()
        print("Resetting EE cold-start stack...")

        # Push latest compose file
        incus_push(COMPOSE_SRC, "/workspace/compose.cold-start.yaml")

        # Tear down (wipes certs and state)
        result = incus_exec(
            "cd /workspace && docker compose -f compose.cold-start.yaml down -v 2>&1",
            timeout=120,
        )
        print(f"  Stack down: {result.stdout.strip()[-200:]}")

        # Clean stale EE friction and checkpoint files
        incus_exec(
            "rm -f /workspace/FRICTION-EE-*.md "
            "/workspace/checkpoint/PROMPT.md "
            "/workspace/checkpoint/RESPONSE.md",
            timeout=15,
        )

        # Write .env — preserve existing ADMIN_PASSWORD/ENCRYPTION_KEY/API_KEY, add licence
        incus_exec(
            "grep -E '^(ADMIN_PASSWORD|ENCRYPTION_KEY|API_KEY|SECRET_KEY)=' "
            "/workspace/.env > /tmp/base.env 2>/dev/null || true; "
            f"printf 'AXIOM_LICENCE_KEY={licence_key}\\n' >> /tmp/base.env; "
            "cp /tmp/base.env /workspace/.env",
            timeout=15,
        )

        # Verify licence key is in .env
        check = incus_exec("grep AXIOM_LICENCE_KEY /workspace/.env | head -1", timeout=10)
        if "AXIOM_LICENCE_KEY" not in check.stdout:
            raise RuntimeError("AXIOM_LICENCE_KEY not written to /workspace/.env")
        print("  Licence key written to /workspace/.env.")

        # Start stack with explicit --env-file to guarantee key reaches agent container
        result = incus_exec(
            "cd /workspace && docker compose --env-file .env "
            "-f compose.cold-start.yaml up -d 2>&1",
            timeout=600,
        )
        print(f"  Stack up: {result.stdout.strip()[-300:]}")

    # ---------------------------------------------------------------------------
    # Node enrollment wait
    # ---------------------------------------------------------------------------

    def get_admin_token() -> str:
        """Get admin JWT from the running EE stack."""
        admin_pw_result = incus_exec(
            "grep '^ADMIN_PASSWORD=' /workspace/.env | cut -d= -f2-", timeout=10
        )
        admin_pw = admin_pw_result.stdout.strip()
        if not admin_pw:
            raise RuntimeError("Could not read ADMIN_PASSWORD from /workspace/.env")

        result = incus_exec(
            f"curl -k -s -X POST https://172.17.0.1:8001/auth/login "
            f"-d 'username=admin&password={admin_pw}' "
            f"| python3 -c \"import json,sys; d=json.load(sys.stdin); "
            f"print(d.get('access_token',''))\"",
            timeout=30,
        )
        token = result.stdout.strip()
        if not token:
            raise RuntimeError(f"Failed to obtain admin token. Response: {result.stdout}")
        return token

    def wait_for_node_enrollment(admin_token: str, timeout: int = 120) -> bool:
        """Poll GET /api/nodes until at least one node shows CONNECTED status."""
        print("Waiting for node enrollment...")
        deadline = time.time() + timeout
        while time.time() < deadline:
            result = incus_exec(
                f"curl -k -s "
                f"-H 'Authorization: Bearer {admin_token}' "
                f"https://172.17.0.1:8001/api/nodes "
                f"| python3 -c \"import json,sys; "
                f"nodes=json.load(sys.stdin); "
                f"connected=[n for n in nodes if n.get('status')=='CONNECTED']; "
                f"print(len(connected))\"",
                timeout=15,
            )
            count = result.stdout.strip()
            if count.isdigit() and int(count) >= 1:
                print(f"  {count} node(s) enrolled and CONNECTED.")
                return True
            print(f"  No nodes connected yet, waiting 10s...")
            time.sleep(10)
        return False

    # ---------------------------------------------------------------------------
    # EE feature verification
    # ---------------------------------------------------------------------------

    def verify_ee_active(admin_token: str) -> bool:
        """Confirm EE is active: /api/features all-true, /api/licence edition=enterprise."""
        print("Verifying EE activation...")

        features_result = incus_exec(
            "curl -k -s https://172.17.0.1:8001/api/features "
            "| python3 -c \"import json,sys; d=json.load(sys.stdin); "
            "ok=all(d.values()); print('ALL_TRUE' if ok else 'PARTIAL'); print(d)\"",
            timeout=15,
        )
        if "ALL_TRUE" not in features_result.stdout:
            print(f"  /api/features not all-true: {features_result.stdout}")
            return False
        print("  /api/features: ALL_TRUE")

        licence_result = incus_exec(
            f"curl -k -s -H 'Authorization: Bearer {admin_token}' "
            f"https://172.17.0.1:8001/api/licence "
            f"| python3 -c \"import json,sys; d=json.load(sys.stdin); "
            f"print('ENTERPRISE' if d.get('edition')=='enterprise' else 'CE'); print(d)\"",
            timeout=15,
        )
        if "ENTERPRISE" not in licence_result.stdout:
            print(f"  /api/licence not enterprise: {licence_result.stdout}")
            return False
        print("  /api/licence: ENTERPRISE")
        return True

    # ---------------------------------------------------------------------------
    # FRICTION file pull
    # ---------------------------------------------------------------------------

    def pull_ee_friction(scenario_id: str, reports_dir: str = REPORTS_DIR) -> bool:
        """Pull FRICTION-EE-{scenario_id}.md from LXC to host reports directory."""
        container_path = f"/workspace/FRICTION-EE-{scenario_id}.md"
        local_path = f"{reports_dir}/FRICTION-EE-{scenario_id}.md"
        success = incus_pull(container_path, local_path)
        if success:
            print(f"  FRICTION-EE-{scenario_id}.md pulled to {local_path}")
        else:
            print(f"  FAILED to pull FRICTION-EE-{scenario_id}.md (file may not exist yet)")
        return success

    # ---------------------------------------------------------------------------
    # CE-gating confirmation
    # ---------------------------------------------------------------------------

    def confirm_ce_gating() -> bool:
        """
        Remove licence key, restart agent only, confirm /api/executions returns 402.

        Restarts only the agent container (not full stack) so enrolled nodes stay connected.
        Re-logins after restart to get a fresh token for the API check.
        """
        print("CE-gating confirmation: removing licence key and restarting agent...")

        # Remove licence key from .env
        incus_exec("sed -i '/^AXIOM_LICENCE_KEY=/d' /workspace/.env", timeout=10)

        # Restart only the agent container
        result = incus_exec(
            "cd /workspace && docker compose --env-file .env "
            "-f compose.cold-start.yaml restart agent",
            timeout=60,
        )
        print(f"  Agent restart: {result.stdout.strip()}")
        time.sleep(15)  # agent startup

        # Re-login for a fresh token (restart does NOT wipe DB/postgres volume)
        try:
            fresh_token = get_admin_token()
        except RuntimeError as e:
            print(f"  Re-login failed after CE gating restart: {e}")
            return False

        # Check /api/executions — expect 402
        result = incus_exec(
            f"curl -k -s -o /dev/null -w '%{{http_code}}' "
            f"-H 'Authorization: Bearer {fresh_token}' "
            f"https://172.17.0.1:8001/api/executions",
            timeout=15,
        )
        status = result.stdout.strip()
        if status == "402":
            print("  CE gating CONFIRMED: /api/executions returns 402 without licence key.")
            return True
        else:
            print(f"  CE gating FAILED: /api/executions returned {status} (expected 402).")
            return False

    # ---------------------------------------------------------------------------
    # Standalone: rebuild, push, reset, verify
    # ---------------------------------------------------------------------------

    if __name__ == "__main__":
        print("=== Phase 64 EE Cold-Start Setup ===")

        # 1. Build EE server image on host
        build_ee_server_image()

        # 2. Smoke-check on host (not in LXC yet)
        print("Smoke-checking EE plugin on host image...")
        smoke_host = subprocess.run(
            ["docker", "run", "--rm", SERVER_IMAGE,
             "python", "-c", "import ee.plugin; print('EE OK')"],
            capture_output=True, text=True, timeout=30,
        )
        if "EE OK" not in smoke_host.stdout:
            print(f"FATAL: EE plugin not in server image. Build output above.")
            sys.exit(1)
        print("  Host smoke check PASSED.")

        # 3. Push to LXC
        push_ee_image_to_lxc()

        # 4. Smoke-check inside LXC
        smoke_check_ee_image()

        # 5. Reset EE stack with licence key
        reset_stack_ee()

        # 6. Wait for stack readiness
        if not wait_for_stack(timeout=300):
            print("FATAL: Stack did not become ready in 5 minutes.")
            sys.exit(1)
        print("  Stack ready.")

        # 7. Get admin token
        token = get_admin_token()

        # 8. Wait for node enrollment
        if not wait_for_node_enrollment(token, timeout=120):
            print("WARNING: No nodes enrolled within 2 minutes — scenarios may fail.")

        # 9. Verify EE active
        if not verify_ee_active(token):
            print("FATAL: EE not active after stack start. Check licence key and image build.")
            sys.exit(1)

        print("=== EE stack ready. Proceed to Plan 02 (ee-install scenario). ===")
    ```
