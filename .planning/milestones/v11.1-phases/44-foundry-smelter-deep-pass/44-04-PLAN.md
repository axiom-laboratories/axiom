iptables isolation pattern (always in try/finally — CRITICAL):
```python
IPTABLES_RULES = [
    ["sudo", "iptables", "-I", "OUTPUT", "-d", "pypi.org", "-j", "DROP"],
    ["sudo", "iptables", "-I", "OUTPUT", "-d", "files.pythonhosted.org", "-j", "DROP"],
]
IPTABLES_REMOVE = [
    ["sudo", "iptables", "-D", "OUTPUT", "-d", "pypi.org", "-j", "DROP"],
    ["sudo", "iptables", "-D", "OUTPUT", "-d", "files.pythonhosted.org", "-j", "DROP"],
]

def add_iptables_rules():
    for rule in IPTABLES_RULES:
        subprocess.run(rule, check=True, timeout=10)

def remove_iptables_rules():
    for rule in IPTABLES_REMOVE:
        # Use check=False — if rule wasn't added, -D will fail with exit 1 but we must not crash
        subprocess.run(rule, check=False, timeout=10)
```
