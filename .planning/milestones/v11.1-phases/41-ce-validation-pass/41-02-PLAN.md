Ed25519 inline signing (CEV-03):
```python
from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PrivateKey
import base64

def sign_script(script: str, key_path: Path) -> str:
    key_pem = key_path.read_bytes()
    private_key = serialization.load_pem_private_key(key_pem, password=None)
    assert isinstance(private_key, Ed25519PrivateKey), (
        f"Expected Ed25519 key at {key_path} but got {type(private_key).__name__}. "
        "The project uses Ed25519 for job signing. Old RSA keys are not valid."
    )
    sig_bytes = private_key.sign(script.encode("utf-8"))  # NO hash arg for Ed25519
    return base64.b64encode(sig_bytes).decode("ascii")
```

```python
def check_signing_key_registered(base_url: str, jwt: str) -> str | None:
    """Return signature ID of any registered key, or None. Matches any key (typically only one)."""
    resp = requests.get(
        f"{base_url}/signatures",   # NOTE: no /api/ prefix
        headers={"Authorization": f"Bearer {jwt}"},
        verify=False, timeout=10,
    )
    if resp.status_code != 200:
        return None
    sigs = resp.json()
    if not sigs:
        return None
    # Return first registered key's ID (print all names to help operator if multiple)
    for s in sigs:
        print(f"  Registered key: name={s.get('name')!r}, id={s.get('id')!r}")
    return sigs[0].get("id")
```

```python
while time.time() < deadline:
    resp = requests.get(
        f"{BASE_URL}/api/executions",
        params={"job_guid": job_guid},
        headers={"Authorization": f"Bearer {jwt}"},
        verify=False, timeout=10,
    )
    records = resp.json()
    if records:
        rec = records[0]
        if rec.get("status") == "COMPLETED" and rec.get("stdout") is not None:
            break
        if rec.get("status") == "FAILED":
            break  # stop polling, will assert below
```
