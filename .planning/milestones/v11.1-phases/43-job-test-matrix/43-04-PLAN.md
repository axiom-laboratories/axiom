Polling until DEAD_LETTER (JOB-07):
```python
# max_retries=2 means 3 total attempts; final status after all retries = DEAD_LETTER
# Poll timeout must be large enough: 3 attempts × ~5s each + retry delays
_TERMINAL = {"COMPLETED", "FAILED", "DEAD_LETTER", "SECURITY_REJECTED"}

def poll_until_terminal(base_url, jwt, job_guid, timeout=120, interval=3):
    deadline = time.time() + timeout
    while time.time() < deadline:
        resp = requests.get(f"{base_url}/api/executions",
                            params={"job_guid": job_guid},
                            headers={"Authorization": f"Bearer {jwt}"},
                            verify=False, timeout=10)
        records = resp.json()
        # Also check job-level status for DEAD_LETTER
        job_resp = requests.get(f"{base_url}/jobs/{job_guid}",
                                headers={"Authorization": f"Bearer {jwt}"},
                                verify=False, timeout=10)
        # Use execution records to detect terminal state
        if records:
            statuses = {r.get("status") for r in records}
            if statuses & _TERMINAL:
                return records
        time.sleep(interval)
    return None
```

DB mutation for JOB-08:
```python
import subprocess

def find_postgres_container():
    result = subprocess.run(
        ["docker", "ps", "--filter", "name=postgres", "--format", "{{.Names}}"],
        capture_output=True, text=True, timeout=10
    )
    names = [n.strip() for n in result.stdout.strip().splitlines() if n.strip()]
    return names[0] if names else None

def corrupt_signature(pg_container, job_def_id):
    result = subprocess.run(
        ["docker", "exec", pg_container,
         "psql", "-U", "postgres", "-d", "postgres",
         "-c", f"UPDATE scheduled_jobs SET signature_payload='INVALIDSIG==' WHERE id='{job_def_id}'"],
        capture_output=True, text=True, timeout=15
    )
    return result.returncode == 0
```
