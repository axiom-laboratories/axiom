```python
def check_scaf04_scenarios():
    """Verify all 4 scenario scripts exist and have required sections."""
    scenarios_dir = Path(__file__).parent.parent / "scenarios"
    required_scenarios = ["ce-install.md", "ce-operator.md", "ee-install.md", "ee-operator.md"]
    required_sections = ["Pass/Fail Checklist", "Checkpoint Trigger Conditions", "Output"]

    results = []
    for scenario in required_scenarios:
        path = scenarios_dir / scenario
        exists = path.exists()
        results.append((f"File exists: {scenario}", exists))
        if exists:
            content = path.read_text()
            for section in required_sections:
                has_section = section in content
                results.append((f"{scenario}: has '{section}'", has_section))
            has_gemini_ref = "gemini-context/GEMINI.md" in content
            results.append((f"{scenario}: references tester GEMINI.md", has_gemini_ref))

    for label, passed in results:
        status = "PASS" if passed else "FAIL"
        print(f"  [{status}] [SCAF-04] {label}")

    return all(ok for _, ok in results)
```
