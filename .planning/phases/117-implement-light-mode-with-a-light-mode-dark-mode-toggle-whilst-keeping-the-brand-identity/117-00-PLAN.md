   import { renderHook, act } from '@testing-library/react';
   import { useTheme } from '../useTheme';

   describe('useTheme hook', () => {
     beforeEach(() => {
       localStorage.clear();
       document.documentElement.classList.remove('dark');
     });

     it('should default to dark theme when localStorage is empty', () => {
       const { result } = renderHook(() => useTheme());
       expect(result.current.theme).toBe('dark');
     });

     it('should read theme from localStorage.mop_theme on init', () => {
       localStorage.setItem('mop_theme', 'light');
       const { result } = renderHook(() => useTheme());
       expect(result.current.theme).toBe('light');
     });

     it('should persist theme to localStorage on setTheme', () => {
       const { result } = renderHook(() => useTheme());
       act(() => {
         result.current.setTheme('light');
       });
       expect(localStorage.getItem('mop_theme')).toBe('light');
     });

     it('should add dark class to document.documentElement when theme is dark', () => {
       const { result } = renderHook(() => useTheme());
       act(() => {
         result.current.setTheme('dark');
       });
       expect(document.documentElement.classList.contains('dark')).toBe(true);
     });

     it('should remove dark class when theme is light', () => {
       document.documentElement.classList.add('dark');
       const { result } = renderHook(() => useTheme());
       act(() => {
         result.current.setTheme('light');
       });
       expect(document.documentElement.classList.contains('dark')).toBe(false);

import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { renderHook, act } from '@testing-library/react';
import { useTheme } from '../hooks/useTheme';
import '../index.css'; // Import CSS to load variables

describe('Theme CSS Variables Integration', () => {
  beforeEach(() => {
    document.documentElement.classList.remove('dark');
    localStorage.clear();
  });

  afterEach(() => {
    document.documentElement.classList.remove('dark');
  });

  it('should have light mode CSS variables in :root scope', () => {
    const root = document.documentElement;
    const bgColor = getComputedStyle(root).getPropertyValue('--background').trim();
    // In light mode (default), --background should be stone-50 equivalent (280 5% 97%)
    expect(bgColor).toBeTruthy();
  });

  it('should have dark mode CSS variables in .dark scope', () => {
    document.documentElement.classList.add('dark');
    const root = document.documentElement;
    const bgColor = getComputedStyle(root).getPropertyValue('--background').trim();
    // In dark mode, --background should be darker value
    expect(bgColor).toBeTruthy();
  });

  it('should apply light background color when theme is light', () => {
    const root = document.documentElement;
    const lightBg = getComputedStyle(root).getPropertyValue('--background');
    expect(lightBg).toBeTruthy();
    // Value should NOT contain 09090b or 121214 (dark values)
    expect(lightBg).not.toContain('09090b');
  });

  it('should apply dark background color when dark class is present', () => {
    document.documentElement.classList.add('dark');
    const root = document.documentElement;
    const darkBg = getComputedStyle(root).getPropertyValue('--background');
    expect(darkBg).toBeTruthy();
  });

  it('should have primary color unchanged in both themes', () => {
    const rootLight = getComputedStyle(document.documentElement).getPropertyValue('--primary').trim();

    document.documentElement.classList.add('dark');
    const rootDark = getComputedStyle(document.documentElement).getPropertyValue('--primary').trim();

    // Primary should be pink (346.8 77.2% 49.8%) in both modes
    expect(rootLight).toContain('346');
    expect(rootDark).toContain('346');
  });

  it('should define status badge colors in CSS variables', () => {
    const root = document.documentElement;
    // Should have variables for success, error, warning badge styles
    const vars = getComputedStyle(root);
    expect(vars.getPropertyValue('--status-success-bg')).toBeTruthy();
    expect(vars.getPropertyValue('--status-error-bg')).toBeTruthy();
    expect(vars.getPropertyValue('--status-warning-bg')).toBeTruthy();
  });

  it('should define shadow CSS variables for light mode cards', () => {
    const root = document.documentElement;
    const vars = getComputedStyle(root);
    expect(vars.getPropertyValue('--shadow')).toBeTruthy();
    expect(vars.getPropertyValue('--shadow-md')).toBeTruthy();
  });
});

import asyncio
from playwright.async_api import async_playwright, expect

BASE_URL = "https://localhost:8443"  # Docker stack Caddy endpoint
ADMIN_USER = "admin"
ADMIN_PASS = os.getenv("ADMIN_PASSWORD", "admin123")

async def test_theme_fowt_prevention():
    """Test that FOWT prevention script prevents flash on reload."""
    async with async_playwright() as p:
        browser = await p.chromium.launch(args=['--no-sandbox'], headless=True)
        page = await browser.new_page()

        # Navigate and log in
        await page.goto(f"{BASE_URL}/login")
        await page.fill('input[name="email"]', ADMIN_USER)
        await page.fill('input[name="password"]', ADMIN_PASS)
        await page.click('button[type="submit"]')
        await page.wait_for_url(f"{BASE_URL}/dashboard")

        # Set light mode
        await page.evaluate("localStorage.setItem('mop_theme', 'light')")
        await page.reload()

        # Check that page loads in light mode immediately (no dark flash)
        # If FOWT prevention works, html.light class should be set before rendering
        light_class = await page.evaluate("document.documentElement.classList.contains('light')")
        assert light_class, "Light class not applied on reload — FOWT prevention failed"

        await browser.close()

async def test_theme_persistence():
    """Test that theme persists across page refresh and sessions."""
    async with async_playwright() as p:
        browser = await p.chromium.launch(args=['--no-sandbox'], headless=True)
        context = await browser.new_context()
        page = await context.new_page()

        # Log in
        await page.goto(f"{BASE_URL}/login")
        await page.fill('input[name="email"]', ADMIN_USER)
        await page.fill('input[name="password"]', ADMIN_PASS)
        await page.click('button[type="submit"]')
        await page.wait_for_url(f"{BASE_URL}/dashboard")

        # Toggle to light mode
        theme_toggle = page.locator('[aria-label*="Switch to"]')
        await theme_toggle.click()
        light_class = await page.evaluate("document.documentElement.classList.contains('light')")
        assert light_class, "Light mode not activated after toggle"

        # Refresh page
        await page.reload()
        light_class_after_reload = await page.evaluate("document.documentElement.classList.contains('light')")
        assert light_class_after_reload, "Light mode not persisted after reload"

        # Close and reopen (new session)
        await context.close()
        context2 = await browser.new_context()
        page2 = await context2.new_page()
        await page2.goto(f"{BASE_URL}/dashboard")
        # Should still have light mode from localStorage
        light_class_new_session = await page2.evaluate("localStorage.getItem('mop_theme')")
        assert light_class_new_session == 'light', "Theme not persisted across sessions"

        await browser.close()

async def test_light_mode_colors():
    """Test that light mode renders with correct CSS variable colors."""
    async with async_playwright() as p:
        browser = await p.chromium.launch(args=['--no-sandbox'], headless=True)
        page = await browser.new_page()

        await page.goto(f"{BASE_URL}/login")
        await page.fill('input[name="email"]', ADMIN_USER)
        await page.fill('input[name="password"]', ADMIN_PASS)
        await page.click('button[type="submit"]')
        await page.wait_for_url(f"{BASE_URL}/dashboard")

        # Set light mode
        await page.evaluate("localStorage.setItem('mop_theme', 'light'); document.documentElement.classList.remove('dark')")
        await page.reload()

        # Check that main layout has light background
        main_bg = await page.evaluate(
            "window.getComputedStyle(document.body).backgroundColor"
        )
        # Light mode background should be stone-50 (light gray, not dark)
        # RGB values for stone-50: rgb(250, 248, 247)
        assert 'rgb(250' in main_bg or 'rgb(245' in main_bg, f"Light mode background incorrect: {main_bg}"

        await browser.close()

async def test_dark_mode_no_regression():
    """Test that dark mode still works and hasn't regressed."""
    async with async_playwright() as p:
        browser = await p.chromium.launch(args=['--no-sandbox'], headless=True)
