**New pattern** (after):
```python
@app.lifespan
async def lifespan(app: FastAPI):
    # Startup
    # Step 1: Run Alembic migrations
    async with engine.begin() as conn:
        await conn.run_sync(run_alembic_upgrade)
    
    # Step 2: Initialize DB (create any missing tables not covered by migrations)
    await init_db()
    
    yield
    
    # Shutdown
    await engine.dispose()

def run_alembic_upgrade(connection):
    """Run Alembic upgrade synchronously within an async context."""
    from alembic.config import Config
    from alembic.script import ScriptDirectory
    from alembic.runtime.migration import MigrationContext
    from alembic.operations import Operations
    
    cfg = Config("alembic.ini")
    # Override database URL from environment if needed
    if DATABASE_URL:
        cfg.set_main_option("sqlalchemy.url", str(DATABASE_URL))
    
    script = ScriptDirectory.from_config(cfg)
    migration_context = MigrationContext.configure(
        connection, 
        opts={"target_metadata": None}  # Don't use target_metadata (autogenerate would be triggered)
    )
    
    with Operations.context(migration_context):
        script.run_env()  # This runs the current alembic.ini configuration
        # Or: script.get_revision(script.get_heads()[0])
        # Or: import alembic.command; alembic.command.upgrade(cfg, "head")
```
