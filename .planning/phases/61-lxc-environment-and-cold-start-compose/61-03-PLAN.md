```python
#!/usr/bin/env python3
"""
Generate a 1-year EE test licence for cold-start validation and write it to secrets.env.

Usage:
    python3 mop_validation/scripts/generate_coldstart_licence.py

Prerequisites:
    - mop_validation/secrets/ee/ee_test_private.pem must exist
      (run generate_ee_keypair.py first if it doesn't)
    - pip install cryptography

Output:
    Upserts AXIOM_EE_LICENCE_KEY=<signed_licence> in mop_validation/secrets.env

The licence is valid for 1 year. Re-running the script regenerates it with a fresh
1-year expiry (idempotent: updates the existing line rather than appending a duplicate).
"""
import base64
import json
import re
import sys
import time
from pathlib import Path

from cryptography.hazmat.primitives.serialization import load_pem_private_key

ROOT = Path(__file__).resolve().parents[2]   # .../Development/
VALIDATION_DIR = ROOT / "mop_validation"
EE_SECRETS_DIR = VALIDATION_DIR / "secrets" / "ee"
SECRETS_ENV_PATH = VALIDATION_DIR / "secrets.env"

EE_FEATURES = [
    "foundry", "audit", "webhooks", "triggers",
    "rbac", "resource_limits", "service_principals", "api_keys",
]


def _b64url_encode(data: bytes) -> str:
    return base64.urlsafe_b64encode(data).rstrip(b"=").decode()


def make_licence_key(private_key, payload_dict: dict) -> str:
    payload_bytes = json.dumps(payload_dict, separators=(",", ":")).encode()
    sig_bytes = private_key.sign(payload_bytes)
    return f"{_b64url_encode(payload_bytes)}.{_b64url_encode(sig_bytes)}"


def upsert_secrets_env(secrets_env_path: Path, key: str, value: str) -> None:
    """Update or append KEY=value in secrets.env. Idempotent."""
    content = secrets_env_path.read_text() if secrets_env_path.exists() else ""
    line = f"{key}={value}"
    if re.search(rf"^{re.escape(key)}=", content, re.MULTILINE):
        content = re.sub(rf"^{re.escape(key)}=.*$", line, content, flags=re.MULTILINE)
    else:
        if content and not content.endswith("\n"):
            content += "\n"
        content += line + "\n"
    secrets_env_path.write_text(content)


def main() -> None:
    private_key_path = EE_SECRETS_DIR / "ee_test_private.pem"
    if not private_key_path.exists():
        print(f"[ERROR] Private key not found: {private_key_path}")
        print("Run: python3 mop_validation/scripts/generate_ee_keypair.py")
        sys.exit(1)

    pem_bytes = private_key_path.read_bytes()
    private_key = load_pem_private_key(pem_bytes, password=None)

    payload = {
        "customer_id": "axiom-coldstart-test",
        "exp": int(time.time()) + 365 * 86400,  # 1 year
        "features": EE_FEATURES,
    }
    licence_key = make_licence_key(private_key, payload)

    upsert_secrets_env(SECRETS_ENV_PATH, "AXIOM_EE_LICENCE_KEY", licence_key)

    print(f"[DONE] AXIOM_EE_LICENCE_KEY written to {SECRETS_ENV_PATH}")
    exp_str = time.strftime("%Y-%m-%d", time.gmtime(payload["exp"]))
    print(f"       Expiry: {exp_str}")
    print(f"       Customer: {payload['customer_id']}")


if __name__ == "__main__":
    main()
```
