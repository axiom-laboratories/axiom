```python
def run_in_lxc(container: str, cmd: str, timeout: int = 120) -> None:
    result = subprocess.run(["incus", "exec", container, "--", "bash", "-c", cmd],
                            capture_output=False, timeout=timeout)
    if result.returncode != 0:
        raise RuntimeError(f"Command failed in {container}: {cmd}")
```

Script behaviour:
1. **Check for existing container**: If `incus info axiom-coldstart` succeeds, print
   "[INFO] axiom-coldstart already exists — skipping launch" and skip to step 4.
   If container exists but is stopped, start it with `incus start axiom-coldstart`.

   ```
   incus launch images:ubuntu/24.04 axiom-coldstart
     --config security.nesting=true
     --config raw.apparmor=pivot_root,
   ```
