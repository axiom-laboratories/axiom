class VaultConfigResponse(BaseModel):
    """Response for GET /admin/vault/config. Masks secret_id for security."""
    vault_address: str
    role_id: str
    secret_id_masked: str = Field(description="First 8 chars of secret_id")
    mount_path: str
    namespace: Optional[str] = None
    provider_type: str

    created_at: datetime

async def dispatch_job(job_id: str, node: Node, db: AsyncSession, vault_service=None) -> WorkResponse:
    """Dispatch a job to a node, optionally resolving Vault secrets."""
    
    # [EXISTING CODE: fetch job from DB, validate, etc.]
    
    # NEW: Resolve Vault secrets if requested
    injected_env = {}
    if getattr(job_req, 'use_vault_secrets', False) and getattr(job_req, 'vault_secrets', None):
        if vault_service is None:
            from fastapi import HTTPException
            raise HTTPException(
                status_code=422,
                detail={
                    "error": "vault_not_configured",
                    "message": "Vault is not available for this deployment",
                }
            )
        
        vault_status = vault_service.status()
        if vault_status != "healthy":
            raise HTTPException(
                status_code=422,
                detail={
                    "error": "vault_unavailable",
                    "vault_status": vault_status,
                    "message": f"Vault unavailable (status: {vault_status}) — cannot resolve secrets for this job",
                }
            )
        
        try:
            secrets_dict = await vault_service.resolve(job_req.vault_secrets)
            # Inject as VAULT_SECRET_<NAME> env vars
            for secret_name, secret_value in secrets_dict.items():
                injected_env[f"VAULT_SECRET_{secret_name}"] = secret_value
        except Exception as e:
            raise HTTPException(
                status_code=422,
                detail={
                    "error": "secret_resolution_failed",
                    "message": f"Failed to resolve Vault secret: {str(e)}",
                }
            )
    
    # [EXISTING CODE: build WorkResponse]
    work_response = WorkResponse(
        guid=job.guid,
        task_type=job.task_type,
        payload=decrypted_payload,
        # ... existing fields ...
        env_vars={**(existing_env_vars or {}), **injected_env}  # Merge Vault secrets with any existing env vars
    )
    
    return work_response

@app.get("/admin/vault/config", tags=["admin"])
async def get_vault_config(
    current_user = Depends(require_ee()),
    db: AsyncSession = Depends(get_db),
):
    """Retrieve current Vault configuration (with masked secret_id)."""
    from fastapi import HTTPException
    
    stmt = select(VaultConfig)
    result = await db.execute(stmt)
    config = result.scalar_one_or_none()
    
    if not config:
        raise HTTPException(status_code=404, detail="Vault configuration not found")
    
    return VaultConfigResponse.from_vault_config(config)


@app.patch("/admin/vault/config", tags=["admin"])
async def update_vault_config(
    req: VaultConfigUpdateRequest,
    current_user = Depends(require_ee()),
    db: AsyncSession = Depends(get_db),
):
    """Update Vault configuration and reinitialize vault_service."""
    from fastapi import HTTPException
    from agent_service.db import VaultConfig
    from agent_service.security import cipher_suite
    import uuid
    
    global vault_service
    
    stmt = select(VaultConfig)
    result = await db.execute(stmt)
    config = result.scalar_one_or_none()
    
    if not config:
        # Create new config if none exists
        config = VaultConfig(
            id=str(uuid.uuid4()),
            vault_address=req.vault_address or "",
            role_id=req.role_id or "",
            secret_id=cipher_suite.encrypt((req.secret_id or "").encode()).decode(),
            mount_path=req.mount_path or "secret",
            namespace=req.namespace,
            provider_type=req.provider_type or "vault",
            enabled=req.enabled if req.enabled is not None else False,
        )
        db.add(config)
    else:
        # Update existing config (only fields provided in request)
        if req.vault_address is not None:
            config.vault_address = req.vault_address
        if req.role_id is not None:
            config.role_id = req.role_id
        if req.secret_id is not None:
            config.secret_id = cipher_suite.encrypt(req.secret_id.encode()).decode()
        if req.mount_path is not None:
            config.mount_path = req.mount_path
        if req.namespace is not None:
            config.namespace = req.namespace
        if req.provider_type is not None:
            config.provider_type = req.provider_type
        if req.enabled is not None:
            config.enabled = req.enabled
        config.updated_at = datetime.utcnow()
    
    try:
        await db.commit()
        await db.refresh(config)
    except Exception as e:
        await db.rollback()
        raise HTTPException(status_code=500, detail=f"Database error: {str(e)}")
    
    # Reinitialize vault_service with updated config
    vault_service = VaultService(config, db) if config.enabled else VaultService(None, db)
    await vault_service.startup()
    
    audit(
        db,
        action="vault_config_updated",
        resource_type="vault_config",
        resource_id="system",
        user=current_user.username,
        details={"fields_updated": [f for f in req.model_fields if getattr(req, f) is not None]}
    )
    
    return VaultConfigResponse.from_vault_config(config)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → API | User provides Vault credentials (address, role_id, secret_id) over HTTPS |
| API → Database | Encrypted secret_id stored in VaultConfig table |
| API → Vault | AppRole credentials used to authenticate with Vault server over TLS |
| Node ← API | Node receives resolved secrets as env vars (VAULT_SECRET_<NAME>) in WorkResponse |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-167-01 | S (Spoofing) | /admin/vault/config endpoints | mitigate | All routes require EE licence via `require_ee()` check; CE users get 403 |
| T-167-02 | T (Tampering) | VaultConfigUpdateRequest | mitigate | Validate vault_address format (must be valid URL); all fields type-checked by Pydantic |
| T-167-03 | D (Disclosure) | VaultConfigResponse.secret_id | mitigate | Return only first 8 chars + "..." in response; full value never exposed in API responses |
| T-167-04 | D (Disclosure) | secret_id in database | mitigate | Encrypt using Fernet cipher_suite at rest (same pattern as existing secrets) |
| T-167-05 | T (Tampering) | Vault credentials in transit | mitigate | HTTPS only; Vault server TLS verification enabled (verify=True in hvac client) |
| T-167-06 | R (Repudiation) | Config updates | mitigate | Audit log entry created via `audit()` helper for all config changes |
| T-167-07 | I (Information Disclosure) | Error messages | mitigate | Test connection endpoint returns generic error details; no internal stack traces exposed |
| T-167-08 | E (Elevation) | Secret injection | mitigate | Secrets injected as env vars ONLY; never embedded in signed script content (signature remains valid) |

</threat_model>
