@app.get("/admin/vault/status", tags=["admin"])
async def get_vault_status(
    current_user = Depends(require_ee()),
):
    """Detailed Vault connection status. Includes last check time and error details if degraded."""
    global vault_service
    
    if vault_service is None:
        from fastapi import HTTPException
        raise HTTPException(status_code=404, detail="Vault service not configured")
    
    status = vault_service.status()
    
    # Get current Vault config for address
    async with AsyncSessionLocal() as db:
        stmt = select(VaultConfig)
        result = await db.execute(stmt)
        config = result.scalar_one_or_none()
    
    vault_address = config.vault_address if config else "unknown"
    
    # Get renewal failure count from vault_service
    renewal_failures = getattr(vault_service, '_consecutive_renewal_failures', 0)
    
    error_detail = None
    if status == "degraded":
        # Provide a generic error message (don't expose internal details)
        error_detail = "Vault connection unavailable or lease renewal failed"
    
    return VaultStatusResponse(
        status=status,
        vault_address=vault_address,
        last_checked_at=datetime.utcnow(),  # Current check time
        error_detail=error_detail,
        renewal_failures=renewal_failures,
    )
```

This route requires EE licence (via require_ee()) and provides detailed status information that operators can use for monitoring and alerting.

async def _renew_vault_lease():
    """Background task to renew Vault lease every 5 minutes.
    
    Called by APScheduler on interval. Catches exceptions and logs errors
    (does not raise, per D-07 graceful degradation).
    """
    try:
        # Import vault_service from main module (lazy import to avoid circular deps)
        import sys
        main_module = sys.modules.get('agent_service.main')
        if main_module is None or not hasattr(main_module, 'vault_service'):
            # Vault not initialized
            return
        
        vault_service = main_module.vault_service
        if vault_service is None:
            # Vault not configured
            return
        
        # Call vault_service.renew() (async method)
        await vault_service.renew()
        logger.debug("Vault lease renewal completed")
    except Exception as e:
        logger.error(f"Vault lease renewal failed: {str(e)}")
        # Do not raise — graceful degradation means the task failure does not crash the platform
```

**Step 3: Add method to schedule the renewal task in SchedulerService (if class-based) or call from main.py lifespan**

In SchedulerService class, add:
```python
@staticmethod
def schedule_vault_lease_renewal():
    """Schedule background Vault lease renewal task.
    
    Runs every 5 minutes (300 seconds). If vault_service not configured,
    this is a no-op. Task runs at 30% of TTL remaining per D-10.
    """
    # Use job ID to prevent duplicate registration on restart
    job_id = "vault_lease_renewal"
    
    # Check if job already exists (avoid duplicate)
    if scheduler_service.get_job(job_id):
        return
    
    scheduler_service.add_job(
        _renew_vault_lease,
        trigger="interval",
        seconds=300,  # Every 5 minutes
        id=job_id,
        name="Vault Lease Renewal",
        replace_existing=True,  # Replace if already scheduled
    )
    logger.info("Scheduled Vault lease renewal task (every 5 minutes)")
```
