       ```python
       from playwright.sync_api import sync_playwright
       import requests
       import json
       import os
       from datetime import datetime

       BASE_URL = "https://localhost:8001"
       DASHBOARD_URL = f"{BASE_URL}/dashboard"
       ADMIN_PASSWORD = os.getenv('ADMIN_PASSWORD', 'admin123')  # or from secrets.env

       def test_ui_polish():
           with sync_playwright() as p:
               browser = p.chromium.launch(headless=True, args=["--no-sandbox"])
               context = browser.new_context(
                   ignore_https_errors=True,
                   viewport={"width": 1400, "height": 900}
               )
               page = context.new_page()

               # 1. Login via API to get token
               login_res = requests.post(
                   f"{BASE_URL}/api/auth/login",
                   data={"username": "admin", "password": ADMIN_PASSWORD},
                   verify=False
               )
               token = login_res.json()["access_token"]

               # 2. Inject token into localStorage (React controlled inputs don't work with fill())
               page.goto(f"{DASHBOARD_URL}/", timeout=15000)
               page.evaluate(f"localStorage.setItem('mop_auth_token', '{token}')")
               page.reload()
               page.wait_for_load_state("networkidle", timeout=10000)

               # 3. Test routes in both themes
               routes_to_test = [
                   "/dashboard",
                   "/nodes",
                   "/jobs",
                   "/job-definitions",
                   "/templates",
                   "/signatures",
                   "/users",
                   "/audit-log",
                   "/admin",
               ]

               screenshots_dir = os.path.expanduser("~/Development/mop_validation/reports/ui-polish-118/screenshots")
               os.makedirs(screenshots_dir, exist_ok=True)

               results = {
                   "timestamp": datetime.now().isoformat(),
                   "routes": [],
                   "checks": {
                       "no_console_errors": True,
                       "no_layout_overflow": True,
                       "theme_toggle_works": True,
                   }
               }

               # 4. For each route, test in light and dark mode
               for route in routes_to_test:
                   route_results = {"route": route, "light": {}, "dark": {}}

                   # Test light mode
                   page.goto(f"{DASHBOARD_URL}{route}", timeout=15000)
                   page.wait_for_load_state("networkidle", timeout=5000)

                   console_errors_light = collect_console_errors(page)
                   has_overflow_light = check_layout_overflow(page)

                   screenshot_path_light = f"{screenshots_dir}/{route.strip('/').replace('/', '-')}-light.png"
                   page.screenshot(path=screenshot_path_light, full_page=True)

                   route_results["light"] = {
                       "screenshot": screenshot_path_light,
                       "console_errors": console_errors_light,
                       "has_overflow": has_overflow_light,
                       "accessible_names": check_accessible_names(page),
                   }

                   # Switch to dark mode
                   toggle_theme(page)
                   page.wait_for_timeout(500)  # Wait for theme transition

                   console_errors_dark = collect_console_errors(page)
                   has_overflow_dark = check_layout_overflow(page)

                   screenshot_path_dark = f"{screenshots_dir}/{route.strip('/').replace('/', '-')}-dark.png"
                   page.screenshot(path=screenshot_path_dark, full_page=True)

                   route_results["dark"] = {
                       "screenshot": screenshot_path_dark,
                       "console_errors": console_errors_dark,
                       "has_overflow": has_overflow_dark,
                       "accessible_names": check_accessible_names(page),
                   }

                   # Switch back to light for next route
                   toggle_theme(page)
                   page.wait_for_timeout(500)

                   results["routes"].append(route_results)

                   # Update overall checks
                   if console_errors_light or console_errors_dark:
                       results["checks"]["no_console_errors"] = False
                   if has_overflow_light or has_overflow_dark:
                       results["checks"]["no_layout_overflow"] = False

               browser.close()
               return results
       ```

       ```python
       def collect_console_errors(page):
           """Collect console errors from page (filtered for known benign messages)."""
           errors = []
           allowlist = [
               "ResizeObserver loop limit exceeded",  # Common, benign
               "Non-Error promise rejection",  # React internals sometimes
               "Invalid header value",  # Occasional browser quirk
           ]

           # Use page.on('console') listener
           messages = []
           page.on('console', lambda msg: messages.append(msg))

           # Trigger any lazy-loaded content
           page.evaluate("window.scrollBy(0, window.innerHeight * 5);")
           page.wait_for_timeout(1000)
           page.evaluate("window.scrollTo(0, 0);")

           for msg in messages:
               text = msg.text
               if msg.type == 'error' and not any(allowed in text for allowed in allowlist):
                   errors.append(text)

           return errors

       def check_layout_overflow(page):
           """Check if page has horizontal scroll (layout overflow)."""
           # Compare document width to window width
           overflow = page.evaluate("""
               document.body.scrollWidth > window.innerWidth
           """)
           return overflow

       def check_accessible_names(page):
           """Verify interactive elements have accessible names (WCAG AA)."""
           issues = []
           # Check for buttons without text or aria-label
           buttons_without_text = page.evaluate("""
               Array.from(document.querySelectorAll('button')).filter(btn =>
                   !btn.textContent.trim() && !btn.getAttribute('aria-label')
               ).length
           """)
           if buttons_without_text > 0:
               issues.append(f"{buttons_without_text} buttons without accessible names")

           # Check for links without text
           links_without_text = page.evaluate("""
               Array.from(document.querySelectorAll('a')).filter(a =>
                   !a.textContent.trim() && !a.getAttribute('aria-label')
               ).length
           """)
           if links_without_text > 0:
               issues.append(f"{links_without_text} links without accessible names")

           return issues if issues else ["✓ All interactive elements have accessible names"]

       def toggle_theme(page):
           """Click the theme toggle in sidebar."""
           page.click('button[data-test="theme-toggle"]')  # Adjust selector as needed
           # Or use Sonner toast + theme context:
           page.evaluate("""
               const button = Array.from(document.querySelectorAll('button'))
                   .find(b => b.querySelector('svg') && (b.textContent.includes('Sun') || b.textContent.includes('Moon')));
               if (button) button.click();
           """)
       ```

       ```python
       def save_results(results):
           """Save results to JSON and generate markdown report."""
           report_dir = os.path.expanduser("~/Development/mop_validation/reports/ui-polish-118")
           os.makedirs(report_dir, exist_ok=True)

           # Save raw JSON
           with open(f"{report_dir}/quality_checks.json", "w") as f:
               json.dump(results, f, indent=2)

           # Generate markdown report
           report = "# UI Polish Verification Report\n\n"
           report += f"**Generated:** {results['timestamp']}\n\n"
           report += "## Quality Checks Summary\n\n"
           for check, passed in results["checks"].items():
               status = "✅ PASS" if passed else "❌ FAIL"
               report += f"- {check}: {status}\n"

           report += "\n## Routes Tested\n\n"
           for route_result in results["routes"]:
               route = route_result["route"]
               report += f"### {route}\n\n"
               report += "| Theme | Screenshot | Console Errors | Overflow |\n"
               report += "|-------|-----------|--------|----------|\n"
               for theme in ["light", "dark"]:
                   data = route_result[theme]
                   errors_count = len(data["console_errors"])
                   overflow = "❌" if data["has_overflow"] else "✅"
                   errors_badge = f"❌ {errors_count}" if errors_count > 0 else "✅"
                   report += f"| {theme} | [Screenshot]({data['screenshot']}) | {errors_badge} | {overflow} |\n"

               report += "\n"

           with open(f"{report_dir}/report.md", "w") as f:
               f.write(report)

           print(f"✅ Report saved to {report_dir}")
       ```
