```python
# Pattern: /tmp/puppet_build_{tmpl.id}_{md5_hash[:8]}
import subprocess

def glob_build_dirs(agent_container: str = "puppeteer-agent-1") -> list[str]:
    """Find puppet_build_* dirs inside the agent container."""
    result = subprocess.run(
        ["docker", "exec", agent_container,
         "find", "/tmp", "-maxdepth", "1", "-name", "puppet_build_*", "-type", "d"],
        capture_output=True, text=True, timeout=10
    )
    return [l.strip() for l in result.stdout.splitlines() if l.strip()]
```
