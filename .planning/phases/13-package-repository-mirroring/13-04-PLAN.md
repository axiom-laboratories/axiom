```python
async def get_smelter_mirror_health(

    """
    Returns:
    {
      "pypi_online": bool,   # socket check to pypi:8080
      "apt_online": bool,    # socket check to mirror:80
      "disk_used_gb": float, # from shutil.disk_usage(MIRROR_DATA_PATH).used
      "disk_total_gb": float
    }
    """

@app.post("/api/smelter/ingredients/{id}/upload", tags=["Smelter Registry"])
async def upload_smelter_package(
    id: str,
    file: UploadFile = File(...),
    current_user: User = Depends(require_permission("foundry:write")),
    db: AsyncSession = Depends(get_db),
) -> dict:
    """
    Routes by file extension: .deb -> MIRROR_DATA_PATH/apt/, else -> MIRROR_DATA_PATH/pypi/.
    Sets ingredient.mirror_status = "MIRRORED".
    Audits: "smelter:package_uploaded".
    Returns: {"status": "MIRRORED", "filename": file.filename}
    """

@app.get("/api/admin/mirror-config", tags=["Admin"])
async def get_mirror_config(
    current_user: User = Depends(require_permission("admin:read")),
    db: AsyncSession = Depends(get_db),
) -> dict:
    """
    Reads PYPI_MIRROR_URL and APT_MIRROR_URL from the Config DB table.
    Falls back to env vars if not set in DB.
    Returns: {"pypi_mirror_url": str, "apt_mirror_url": str}
    """

@app.put("/api/admin/mirror-config", tags=["Admin"])
async def update_mirror_config(
    payload: dict,  # {"pypi_mirror_url": str, "apt_mirror_url": str}
    current_user: User = Depends(require_permission("admin:write")),
    db: AsyncSession = Depends(get_db),
) -> dict:
    """
    Writes pypi_mirror_url and apt_mirror_url to Config DB table using upsert.
    Audits: "admin:mirror_config_updated".
    Returns: {"status": "updated"}
    Note: Changes take effect for the next build — running builds are not affected.
    """
```

`GET /api/smelter/mirror-health`:
- Requires `foundry:read` permission.
- Creates a local `check_port(host, port)` helper using `socket.create_connection((host, port), timeout=1)`.
- Reads `MIRROR_DATA_PATH` from env (default `/app/mirror_data`); if path exists, runs `shutil.disk_usage(mirror_path)` and records used/total in GB.
- Checks `pypi:8080` (pypi sidecar) and `mirror:80` (Caddy APT sidecar).
- Returns `{"pypi_online": bool, "apt_online": bool, "disk_used_gb": float, "disk_total_gb": float}`.

`POST /api/smelter/ingredients/{id}/upload`:
- Requires `foundry:write` permission.
- Fetches `ApprovedIngredient` by `id`, raises 404 if missing.
- Routes by file extension (NOT os_family):
  ```python
  if file.filename.endswith(".deb"):
      target_dir = os.path.join(mirror_path, "apt")
  else:
      # .whl and everything else goes to pypi
      target_dir = os.path.join(mirror_path, "pypi")
  ```
- Writes file to `target_dir/filename` using `shutil.copyfileobj`.
- Sets `ing.mirror_status = "MIRRORED"` and `ing.mirror_path = target_dir`.
- Calls `audit(db, current_user, "smelter:package_uploaded", f"{ing.name}:{file.filename}")`.
- Returns `{"status": "MIRRORED", "filename": file.filename}`.

`GET /api/admin/mirror-config`:
- Requires `admin:read` permission (or at minimum `foundry:read` if admin:read doesn't exist — use the most restrictive existing admin permission).
- Reads Config rows with keys `PYPI_MIRROR_URL` and `APT_MIRROR_URL` from the DB.
- Falls back to env vars if not found in DB: `os.getenv("PYPI_MIRROR_URL", "http://pypi:8080/simple")` and `os.getenv("APT_MIRROR_URL", "http://mirror/apt")`.
- Returns `{"pypi_mirror_url": str, "apt_mirror_url": str}`.

`PUT /api/admin/mirror-config`:
- Requires admin permission (same as GET above).
- Accepts JSON body `{"pypi_mirror_url": str, "apt_mirror_url": str}`.
- Upserts both values into Config table (SELECT then UPDATE or INSERT).
- Audits: `audit(db, current_user, "admin:mirror_config_updated", f"pypi={payload['pypi_mirror_url']}")`.
- Returns `{"status": "updated"}`.

Ensure `import socket` and `import shutil` are present in the imports at the top of main.py. Also ensure `UploadFile` and `File` are imported from `fastapi`.

```tsx
const uploadMutation = useMutation({
  mutationFn: async ({ id, file }: { id: string; file: File }) => {

    const file = (e.target as HTMLInputElement).files?.[0];
    if (file) uploadMutation.mutate({ id, file });

  input.click();
};
```

```tsx
// Mirror config state and query
const [mirrorConfigEdit, setMirrorConfigEdit] = useState<{pypi_mirror_url: string; apt_mirror_url: string} | null>(null);

const { data: mirrorConfig } = useQuery({
  queryKey: ['mirror-config'],
  queryFn: async () => {
    const res = await authenticatedFetch('/api/admin/mirror-config');
    return res.json();
  },
});

const updateMirrorConfigMutation = useMutation({
  mutationFn: async (cfg: {pypi_mirror_url: string; apt_mirror_url: string}) => {
    const res = await authenticatedFetch('/api/admin/mirror-config', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(cfg),
    });
    if (!res.ok) throw new Error("Failed to update mirror config");
    return res.json();
  },
  onSuccess: () => {
    queryClient.invalidateQueries({ queryKey: ['mirror-config'] });
    toast.success("Mirror sources updated");
    setMirrorConfigEdit(null);
  },
  onError: () => toast.error("Failed to update mirror sources"),
});
```
