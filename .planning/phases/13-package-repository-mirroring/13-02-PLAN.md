Create puppeteer/agent_service/services/mirror_service.py:

```python
import logging
import asyncio
import os
import subprocess
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from ..db import ApprovedIngredient, AsyncSessionLocal

logger = logging.getLogger(__name__)

class MirrorService:
    MIRROR_BASE_PATH = os.getenv("MIRROR_DATA_PATH", "/app/mirror_data")
    PYPI_PATH = os.path.join(MIRROR_BASE_PATH, "pypi")
    APT_PATH = os.path.join(MIRROR_BASE_PATH, "apt")

    @staticmethod
    async def mirror_ingredient(ingredient_id: str):
        """
        Background task. Opens its own DB session (not the request session)
        to avoid conflicts with the caller's transaction lifecycle.
        Sets mirror_status=MIRRORED on success, FAILED on error.
        """
        async with AsyncSessionLocal() as db:
            res = await db.execute(
                select(ApprovedIngredient).where(ApprovedIngredient.id == ingredient_id)
            )
            ingredient = res.scalar_one_or_none()
            if not ingredient:
                logger.error(f"Mirror: ingredient {ingredient_id} not found")
                return

            logger.info(f"Mirror: processing {ingredient.name} ({ingredient.os_family})")
            try:
                await MirrorService._mirror_pypi(db, ingredient)
                ingredient.mirror_status = "MIRRORED"
                await db.commit()
                logger.info(f"Mirror: successfully mirrored {ingredient.name}")
            except Exception as exc:
                logger.error(f"Mirror: failed to mirror {ingredient.name}: {exc}")
                ingredient.mirror_status = "FAILED"
                await db.commit()

    @staticmethod
    async def _mirror_pypi(db: AsyncSession, ingredient: ApprovedIngredient):
        """Downloads a Python package wheel to the local PyPI packages directory."""
        os.makedirs(MirrorService.PYPI_PATH, exist_ok=True)

        req = ingredient.name
        if ingredient.version_constraint:
            req += ingredient.version_constraint

        cmd = [
            "pip", "download",
            "--dest", MirrorService.PYPI_PATH,
            "--no-deps",
            "--platform", "manylinux2014_x86_64",
            "--only-binary=:all:",
            req,
        ]

        process = await asyncio.to_thread(
            subprocess.run, cmd, capture_output=True, text=True
        )

        if process.returncode != 0:
            raise RuntimeError(f"pip download failed: {process.stderr}")

        ingredient.mirror_path = MirrorService.PYPI_PATH

    @staticmethod
    def get_pip_conf_content() -> str:
        """Returns pip.conf pointing to the local PyPI mirror sidecar."""
        url = os.getenv("PYPI_MIRROR_URL", "http://pypi:8080/simple")
        # Extract hostname for trusted-host (strips port and path)
        host = url.split("//")[-1].split(":")[0].split("/")[0]
        return f"[global]\nindex-url = {url}\ntrusted-host = {host}\n"

    @staticmethod
    def get_sources_list_content() -> str:
        """Returns sources.list pointing to the local APT mirror sidecar."""
        url = os.getenv("APT_MIRROR_URL", "http://mirror/apt")
        return f"deb [trusted=yes] {url} stable main\n"
```

import pytest
from unittest.mock import patch, AsyncMock, MagicMock
from puppeteer.agent_service.services.mirror_service import MirrorService
from puppeteer.agent_service.db import ApprovedIngredient


@pytest.mark.asyncio
async def test_mirror_pypi_command_construction():
    """pip download command includes all required flags and the versioned package name."""
    mock_db = AsyncMock()
    ingredient = ApprovedIngredient(
        id="test-id", name="requests", version_constraint="==2.20.0", os_family="DEBIAN"
    )
    with patch("subprocess.run") as mock_run, patch("os.makedirs"):
        mock_run.return_value = MagicMock(returncode=0, stdout="", stderr="")
        await MirrorService._mirror_pypi(mock_db, ingredient)
    cmd = mock_run.call_args[0][0]
    assert "pip" in cmd
    assert "download" in cmd
    assert "--dest" in cmd
    assert "--no-deps" in cmd
    assert "--platform" in cmd
    assert "manylinux2014_x86_64" in cmd
    assert "requests==2.20.0" in cmd


@pytest.mark.asyncio
async def test_mirror_ingredient_success():
    """mirror_ingredient sets mirror_status=MIRRORED and commits on successful download."""
    ingredient = ApprovedIngredient(
        id="test-id", name="flask", version_constraint="==2.0.0",
        os_family="DEBIAN", mirror_status="PENDING"
    )
    with patch("puppeteer.agent_service.services.mirror_service.AsyncSessionLocal") as mock_factory:
        mock_db = AsyncMock()
        mock_factory.return_value.__aenter__.return_value = mock_db
        mock_res = MagicMock()
        mock_res.scalar_one_or_none.return_value = ingredient
        mock_db.execute.return_value = mock_res
        with patch.object(MirrorService, "_mirror_pypi", new_callable=AsyncMock):
            await MirrorService.mirror_ingredient("test-id")
    assert ingredient.mirror_status == "MIRRORED"
    mock_db.commit.assert_called()


@pytest.mark.asyncio
async def test_mirror_ingredient_failure():
    """mirror_ingredient sets mirror_status=FAILED and commits when download raises."""
    ingredient = ApprovedIngredient(
        id="test-id", name="broken-pkg", version_constraint="==1.0.0",
        os_family="DEBIAN", mirror_status="PENDING"
    )
    with patch("puppeteer.agent_service.services.mirror_service.AsyncSessionLocal") as mock_factory:
        mock_db = AsyncMock()
        mock_factory.return_value.__aenter__.return_value = mock_db
        mock_res = MagicMock()
        mock_res.scalar_one_or_none.return_value = ingredient
        mock_db.execute.return_value = mock_res
        with patch.object(MirrorService, "_mirror_pypi", new_callable=AsyncMock,
                          side_effect=RuntimeError("pip download failed")):
            await MirrorService.mirror_ingredient("test-id")
    assert ingredient.mirror_status == "FAILED"
    mock_db.commit.assert_called()


def test_pip_conf_generation():
    """pip.conf contains the PYPI_MIRROR_URL and derived trusted-host."""
    with patch.dict("os.environ", {"PYPI_MIRROR_URL": "http://my-pypi:8080/simple"}):
        content = MirrorService.get_pip_conf_content()
    assert "index-url = http://my-pypi:8080/simple" in content
    assert "trusted-host = my-pypi" in content


def test_sources_list_generation():
    """sources.list uses APT_MIRROR_URL with [trusted=yes] tag."""
    with patch.dict("os.environ", {"APT_MIRROR_URL": "http://apt-mirror/apt"}):
        content = MirrorService.get_sources_list_content()
    assert "deb [trusted=yes] http://apt-mirror/apt stable main" in content
