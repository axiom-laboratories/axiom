Implement src/utils/dagValidation.ts:

**validateDAG(nodes: Node[], edges: Edge[], maxDepth: number = 30): ValidationResult**
1. Build adjacency list: Map<string, string[]> from edges
2. DFS cycle detection using visited + recStack sets:
   - For each unvisited node, run dfs(node, path=[])
   - Mark visited and push to recStack
   - Recurse on all neighbors
   - If neighbor is in recStack, we found a back edge → cycle detected
   - Extract cycle path from current path
3. Calculate max depth (longest path in DAG):
   - Use memoized DFS from each node
   - Return max path length across all nodes
4. Return ValidationResult:
   - isValid = !hasCycle && maxDepth <= maxDepth
   - hasCycle = (back edge found)
   - cycleNodes = [path from cycle start to back edge]
   - maxDepth = longest path
   - depthExceeded = maxDepth > maxDepth

**Key implementation notes:**
- Use TypeScript Node and Edge types from @xyflow/react
- O(V+E) DFS is efficient; cache results
- Test with diamond graphs (A→{B,C}→D) to avoid false positives
- Ignore edges with source/target not in nodes set

import { useState, useEffect } from 'react';
import { Node, Edge } from '@xyflow/react';
import { validateDAG, ValidationResult } from '@/utils/dagValidation';

export function useDAGValidation(nodes: Node[], edges: Edge[]) {
  const [validation, setValidation] = useState<ValidationResult>({
    isValid: true,
    hasCycle: false,
    maxDepth: 0,
    depthExceeded: false,
  });

  useEffect(() => {
    const result = validateDAG(nodes, edges);
    setValidation(result);
  }, [nodes, edges]);

  return {
    validation,
    isValid: validation.isValid,
    hasCycle: validation.hasCycle,
    maxDepth: validation.maxDepth,
    depthExceeded: validation.depthExceeded,
  };
}
```

**Key points:**
- Dependency array: [nodes, edges] ensures recomputation on change
- No debouncing (per spec: real-time validation)
- Returns both full `validation` object and convenience flags
- Used in WorkflowDetail to drive banner display + Save button state

import { useState, useCallback } from 'react';
import { Node, Edge, NodeChange, EdgeChange, Connection } from '@xyflow/react';

export function useWorkflowEdit(initialSteps: Node[], initialEdges: Edge[]) {
  const [nodes, setNodes] = useState<Node[]>(initialSteps);
  const [edges, setEdges] = useState<Edge[]>(initialEdges);
  const [isEditing, setIsEditing] = useState(false);

  const handleNodesChange = useCallback((changes: NodeChange[]) => {
    setNodes((nds) => {
      let updated = [...nds];
      changes.forEach((change) => {
        if (change.type === 'position' && change.position) {
          updated = updated.map((n) =>
            n.id === change.id ? {...n, position: change.position} : n
          );
        } else if (change.type === 'remove') {
          updated = updated.filter((n) => n.id !== change.id);
        } else if (change.type === 'select') {
          updated = updated.map((n) =>
            n.id === change.id
              ? {...n, selected: change.selected}
              : n
          );
        }
      });
      return updated;
    });
  }, []);

  const handleEdgesChange = useCallback((changes: EdgeChange[]) => {
    setEdges((eds) => {
      let updated = [...eds];
      changes.forEach((change) => {
        if (change.type === 'remove') {
          updated = updated.filter((e) => e.id !== change.id);
        } else if (change.type === 'select') {
          updated = updated.map((e) =>
            e.id === change.id
              ? {...e, selected: change.selected}
              : e
          );
        }
      });
      return updated;
    });
  }, []);

  const handleConnect = useCallback((connection: Connection) => {
    if (connection.source && connection.target) {
      const edge: Edge = {
        id: `${connection.source}-${connection.target}`,
        source: connection.source,
        target: connection.target,
      };
      setEdges((eds) => [...eds, edge]);
    }
  }, []);

  const handleDrop = useCallback((nodeType: string, position: {x: number, y: number}) => {
    // Create new node with optional scheduled_job_id
    const newNode: Node = {
      id: `step_${Date.now()}`,
      data: {label: nodeType, nodeType, scheduled_job_id: null},
      position,
      type: 'default',
    };
    setNodes((nds) => [...nds, newNode]);
  }, []);

  const getUnlinkedScriptNodes = useCallback(() => {
    return nodes.filter((n) => n.data.nodeType === 'SCRIPT' && !n.data.scheduled_job_id);
  }, [nodes]);

  const canSave = useCallback(() => {
    return getUnlinkedScriptNodes().length === 0;
  }, [getUnlinkedScriptNodes]);

  return {
    nodes,
    edges,
    isEditing,
    setIsEditing,
    handleNodesChange,
    handleEdgesChange,
    handleConnect,
    handleDrop,
    getUnlinkedScriptNodes,
    canSave,
  };
}
```

**Key points:**
- State: nodes, edges, isEditing
- Handlers apply ReactFlow mutations (position, remove, select)
- handleDrop creates new node at position (position comes from screenToFlowPosition in caller)
- getUnlinkedScriptNodes filters SCRIPT nodes without scheduled_job_id
- canSave checks no unlinked SCRIPT nodes (validation hook checks cycles/depth)

async function handleSave() {
  // Check unlinked SCRIPT nodes
  if (!canSaveHook()) {
    showToast({title: 'Error', description: 'All SCRIPT steps must be linked to a job'});
    return;
  }
  
  // Check validation

    showToast({title: 'Error', description: 'Fix validation errors before saving'});
    return;
  }
  
  // Call backend validate endpoint
  try {
    const validateRes = await authenticatedFetch('/api/workflows/validate', {
      method: 'POST',
      body: JSON.stringify({steps: nodes, edges}),
    });
    const validateResult = await validateRes.json();
    
    if (!validateResult.valid) {
      showToast({title: 'Validation Error', description: validateResult.error});
      return;
    }
  } catch (err) {
    showToast({title: 'Error', description: 'Failed to validate workflow'});
    return;
  }
  
  // Call backend PUT endpoint
  try {
    const putRes = await authenticatedFetch(`/api/workflows/${workflowId}`, {
      method: 'PUT',
      body: JSON.stringify({steps: nodes, edges}),
    });
    if (!putRes.ok) throw new Error('Failed to save');
    
    showToast({title: 'Success', description: 'Workflow saved'});
    setIsEditing(false);
    queryClient.invalidateQueries({queryKey: ['workflow', workflowId]});
  } catch (err) {
    showToast({title: 'Error', description: 'Failed to save workflow'});
  }
}
