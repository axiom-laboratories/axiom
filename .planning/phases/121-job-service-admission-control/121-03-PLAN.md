4. Handling pattern (sketch):
```python
async def fire(scheduled_job_id: str, db: AsyncSession):
    scheduled_job = await db.get(ScheduledJob, scheduled_job_id)

    job_req = JobCreate(
        # ... fields ...
        memory_limit=scheduled_job.memory_limit,
        cpu_limit=scheduled_job.cpu_limit,
    )

    try:
        job_dict = await job_service.create_job(job_req, db)
        return job_dict
    except HTTPException as e:
        if e.status_code == 422:
            # Admission check failed
            job = Job(
                guid=str(uuid.uuid4()),
                status="FAILED",
                reason="admission_rejected",
                detail=e.detail.get("message", "Admission control rejected job"),
                scheduled_job_id=scheduled_job_id,
                # ... other required fields ...
            )
            db.add(job)
            await db.commit()
            logger.warning(f"Scheduled job {scheduled_job_id} admission rejected: {e.detail}")
            return job.to_dict()
        else:
            raise
```

3. Add state and fetch logic:
```tsx
const [diagnosis, setDiagnosis] = useState(null);
const [diagnosisLoading, setDiagnosisLoading] = useState(false);

// Fetch diagnosis when job detail expands or guid changes
useEffect(() => {
  if (job.status === "PENDING") {
    setDiagnosisLoading(true);
    fetch(`/api/jobs/${job.guid}/diagnosis`)
      .then((res) => res.json())
      .then((data) => {
        setDiagnosis(data);
        setDiagnosisLoading(false);
      })
      .catch((err) => {
        console.error("Failed to fetch diagnosis:", err);
        setDiagnosisLoading(false);
      });
  }
}, [job.guid, job.status]);

// Auto-refresh diagnosis on WebSocket job update
useWebSocket(
  (event) => {
    if (event.type === "job_updated" && event.guid === job.guid) {
      if (event.status === "PENDING") {
        // Refetch diagnosis
        fetch(`/api/jobs/${event.guid}/diagnosis`)
          .then((res) => res.json())
          .then((data) => setDiagnosis(data));
      }
    }
  },
  [job.guid]
);
```
