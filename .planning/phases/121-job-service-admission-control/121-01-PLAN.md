## Reference Code to Port

Source: `puppets/environment_service/node.py:25-34` — parse_bytes() function proven in production:
```python
def parse_bytes(s: str) -> int:
    """Convert memory string like '300m', '2g', '1024k' to bytes."""
    s = s.strip().lower()
    if s.endswith('g'):
        return int(s[:-1]) * 1024 ** 3
    elif s.endswith('m'):
        return int(s[:-1]) * 1024 ** 2
    elif s.endswith('k'):
        return int(s[:-1]) * 1024
    return int(s)  # Assume bytes if no suffix
```

    # ... existing validation ...

    # NEW: admission check
    if job_req.memory_limit is not None:
        effective_memory = job_req.memory_limit
    else:
        # Get default from Config table
        config_result = await db.execute(
            select(Config.value).where(Config.key == 'default_job_memory_limit')
        )
        effective_memory = config_result.scalar_one_or_none() or "512m"

    # Check online nodes
    online_result = await db.execute(
        select(Node).where(Node.status.in_(["ONLINE", "BUSY"]))
    )
    online_nodes = online_result.scalars().all()

    if online_nodes:
        job_bytes = parse_bytes(effective_memory)
