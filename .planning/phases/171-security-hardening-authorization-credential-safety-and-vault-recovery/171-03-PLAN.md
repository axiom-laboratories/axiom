Replace with narrowed exception handler:

```python
except (hvac.exceptions.VaultError, hvac.exceptions.InvalidRequest,
        hvac.exceptions.Forbidden, hvac.exceptions.InternalServerError,

Current pattern (lines ~170-180):

async def renew(self):

        logger.debug("Vault not configured or not connected; skipping renewal")

    # ... existing renewal logic
```

Update to:

```python
async def renew(self):
    # Auto re-auth if stuck in degraded state
    if self._status == "degraded" and self.config is not None:
        try:
            await self._connect()
            self._status = "healthy"
            self._consecutive_renewal_failures = 0
            logger.info("Vault re-authentication succeeded — status restored to healthy")
        except Exception as e:
            self._consecutive_renewal_failures += 1
            logger.warning("Vault re-authentication attempt failed: %s", e)
            return  # keep _status = "degraded", don't attempt normal renew
    
    if not self.config or not self._client:
        logger.debug("Vault not configured or not connected; skipping renewal")
        return
    # ... existing renewal logic continues here
```

This means:
1. On renewal timer tick (every 60s from lifespan APScheduler task), if status is degraded, try reconnecting
2. If reconnect succeeds, status becomes healthy and renew continues normally
3. If reconnect fails, increment failure counter (tracks consecutive failures) and bail out
4. No new code needed in the router — the renewal timer becomes the recovery mechanism

Rationale: Transient network blips or token expiry (~1-2 minute outages) now auto-recover on the next renewal cycle without requiring manual config update.

@router.patch(
    "/admin/vault/config/{config_id}",
    response_model=dict,
    tags=["Vault"],
    summary="Update a Vault configuration by ID",
    description="Update specific config by ID (all fields optional). If config is enabled, triggers re-initialization."
)
async def update_vault_config(
    config_id: str,
    req: VaultConfigUpdateRequest,
    current_user: User = Depends(require_ee()),
    db: AsyncSession = Depends(get_db)
):
    """Update config by ID; re-init if enabled."""
    result = await db.execute(select(VaultConfig).where(VaultConfig.id == uuid.UUID(config_id)))
    vault_config = result.scalars().first()
    if not vault_config:
        raise HTTPException(status_code=404, detail="Config not found")
    
    if req.vault_address:
        vault_config.vault_address = req.vault_address
    if req.role_id:
        vault_config.role_id = req.role_id
    if req.secret_id:
        vault_config.secret_id = req.secret_id
    if req.namespace is not None:
        vault_config.namespace = req.namespace
    if req.mount_path:
        vault_config.mount_path = req.mount_path
    
    await db.commit()
    
    # If config is enabled, reinitialize service
    if vault_config.enabled:
        try:
            vault_service = getattr(request.app.state, 'vault_service', None)
            if vault_service:
                vault_service.config = VaultConfigSnapshot(
                    enabled=vault_config.enabled,
                    vault_address=vault_config.vault_address,
                    role_id=vault_config.role_id,
                    secret_id=vault_config.secret_id,
                    mount_path=vault_config.mount_path,
                    namespace=vault_config.namespace,
                    provider_type=vault_config.provider_type,
                )
                await vault_service.startup()
                logger.info("Vault service reinitialized after config update")
        except Exception as e:
            logger.warning(f"Failed to reinitialize Vault service after update: {e}")
    
    return {
        "id": str(vault_config.id),
        "provider_type": vault_config.provider_type,
        "enabled": vault_config.enabled,
        "vault_address": vault_config.vault_address,
        "secret_id": f"****{vault_config.secret_id[-4:]}" if vault_config.secret_id else "****"
    }

@router.post(
    "/admin/vault/config/{config_id}/enable",
    response_model=dict,
    tags=["Vault"],
    summary="Enable this Vault configuration and disable all others",
    description="Switch to this config as active provider. Disables all other configs. Triggers vault_service.startup()."
)
async def enable_vault_config(
    config_id: str,
    current_user: User = Depends(require_ee()),
    db: AsyncSession = Depends(get_db)
):
    """Enable config; disable all others; reinit service."""
    result = await db.execute(select(VaultConfig).where(VaultConfig.id == uuid.UUID(config_id)))
    vault_config = result.scalars().first()
    if not vault_config:

    await db.execute(update(VaultConfig).where(VaultConfig.id != vault_config.id).values(enabled=False))
    vault_config.enabled = True
    await db.commit()
    
    # Reinitialize service with new config
    try:
        vault_service = getattr(request.app.state, 'vault_service', None)
        if vault_service:
            vault_service.config = VaultConfigSnapshot(
                enabled=vault_config.enabled,
                vault_address=vault_config.vault_address,

                secret_id=vault_config.secret_id,
                mount_path=vault_config.mount_path,
                namespace=vault_config.namespace,
                provider_type=vault_config.provider_type,
            )
            await vault_service.startup()
            logger.info(f"Vault service switched to config {config_id}")
    except Exception as e:
        logger.warning(f"Failed to reinitialize Vault service: {e}")
    
    return {
        "id": str(vault_config.id),
        "provider_type": vault_config.provider_type,

| T-171-09 | Elevation of Privilege | vault_service.resolve() exception handler | mitigate | Exception handler narrowed from broad `Exception` to specific `hvac.exceptions.*` + network errors (ConnectionError, TimeoutError, OSError).
