def filter_nodes_by_runtime(all_nodes: List[dict], runtime: str = None) -> Tuple[List[dict], List[dict]]:
    """Filter nodes by execution_mode and cgroup v2; return (passed, skipped)."""
    passed = []
    skipped = []
    for node in all_nodes:
        if runtime and node.get('execution_mode') != runtime:
            skipped.append({"node_id": node['id'], "reason": "execution_mode mismatch", ...})
            continue
        cgroup_ver = node.get('cgroup_version')
        if cgroup_ver != 'v2':
            skipped.append({"node_id": node['id'], "reason": "cgroup_version != v2", "cgroup_version": cgroup_ver})
            continue
        passed.append(node)
    return passed, skipped
