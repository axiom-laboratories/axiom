```python
@pytest.mark.asyncio
async def test_pull_work_includes_limits(async_db_session):
    """Verify that pull_work() populates WorkResponse with memory_limit and cpu_limit from job."""

    # Setup: Create a job with limits
    job_with_limits = Job(
        guid="test-job-limits-01",
        task_type="test",
        payload="{}",
        status="pending",
        memory_limit="512m",
        cpu_limit="2",
    )
    async_db_session.add(job_with_limits)
    await async_db_session.commit()

    # Act: Pull the job
    result = await JobService.pull_work(
        node_id="test-node",
        node_ip="127.0.0.1",
        db=async_db_session,
    )

    # Assert: WorkResponse includes limits
    assert result is not None
    assert result.job.memory_limit == "512m"
    assert result.job.cpu_limit == "2"

    # Verify nullable behavior: job without limits
    job_without_limits = Job(
        guid="test-job-no-limits-01",
        task_type="test",
        payload="{}",
        status="pending",
        memory_limit=None,
        cpu_limit=None,
    )
    async_db_session.add(job_without_limits)
    await async_db_session.commit()

    result_no_limits = await JobService.pull_work(
        node_id="test-node",
        node_ip="127.0.0.1",
        db=async_db_session,
    )

    assert result_no_limits is not None
    assert result_no_limits.job.memory_limit is None
    assert result_no_limits.job.cpu_limit is None
```

**Location:** Add this test after the existing limit format validation tests (after line ~290 in test_job_limits.py).

**Why:** Verifies end-to-end flow: job limit values stored in DB → passed through pull_work() → serialized in WorkResponse → nodes receive values. This closes the gap identified in verification report (lines 739-747 in job_service.py).
