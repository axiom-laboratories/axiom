    Transcribe field definitions exactly from main-branch db.py. Key adaptation rules:
    - Use `from ee.base import EEBase` (not CE's Base)
    - `CapabilityMatrix.artifact_id`: keep as `mapped_column(String, nullable=True)` — DROP the `ForeignKey("artifacts.id")` reference since `Artifact` is in EEBase not CE Base; within-EEBase FKs are allowed (both tables are in EEBase.metadata, so SA can resolve them at create_all time)
    - `ImageBOM.template_id`: keep `ForeignKey` referencing `puppet_templates` — BOTH ImageBOM and PuppetTemplate are in EEBase so the FK is valid
    - `PackageIndex.template_id`: same — both in EEBase, FK is valid
    - No FK to CE tables (users, scheduled_jobs, nodes, etc.)

    ```python
    from __future__ import annotations
    from datetime import datetime
    from typing import Optional
    from uuid import uuid4
    from sqlalchemy import String, Text, Boolean, DateTime, Integer, Float, UniqueConstraint, ForeignKey
    from sqlalchemy.orm import Mapped, mapped_column
    from ee.base import EEBase


    class Blueprint(EEBase):
        __tablename__ = "blueprints"
        id: Mapped[str] = mapped_column(String, primary_key=True, default=lambda: str(uuid4()))
        type: Mapped[str] = mapped_column(String)  # RUNTIME, NETWORK
        name: Mapped[str] = mapped_column(String, unique=True)
        definition: Mapped[str] = mapped_column(Text)  # JSON blob
        version: Mapped[int] = mapped_column(Integer, default=1)
        created_at: Mapped[datetime] = mapped_column(DateTime, default=datetime.utcnow)
        os_family: Mapped[Optional[str]] = mapped_column(String, nullable=True)


    class Artifact(EEBase):
        __tablename__ = "artifacts"
        id: Mapped[str] = mapped_column(String, primary_key=True, default=lambda: str(uuid4()))
        filename: Mapped[str] = mapped_column(String)
        content_type: Mapped[str] = mapped_column(String)
        sha256: Mapped[str] = mapped_column(String)
        size_bytes: Mapped[int] = mapped_column(Integer)
        created_at: Mapped[datetime] = mapped_column(DateTime, default=datetime.utcnow)


    class ApprovedOS(EEBase):
        __tablename__ = "approved_os"
        id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
        name: Mapped[str] = mapped_column(String, unique=True)
        image_uri: Mapped[str] = mapped_column(String)
        os_family: Mapped[str] = mapped_column(String)


    class CapabilityMatrix(EEBase):
        __tablename__ = "capability_matrix"
        id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
        base_os_family: Mapped[str] = mapped_column(String)
        tool_id: Mapped[str] = mapped_column(String)
        injection_recipe: Mapped[str] = mapped_column(Text)
        validation_cmd: Mapped[str] = mapped_column(String)
        artifact_id: Mapped[Optional[str]] = mapped_column(String, nullable=True)
        # No FK("artifacts.id") here — both in EEBase but drop for simplicity per CONTEXT.md
        runtime_dependencies: Mapped[str] = mapped_column(Text, default="[]", server_default="[]")
        is_active: Mapped[bool] = mapped_column(Boolean, default=True, server_default="true")


    class PuppetTemplate(EEBase):
        __tablename__ = "puppet_templates"
        id: Mapped[str] = mapped_column(String, primary_key=True, default=lambda: str(uuid4()))
        friendly_name: Mapped[str] = mapped_column(String, unique=True)
        runtime_blueprint_id: Mapped[str] = mapped_column(String)
        network_blueprint_id: Mapped[str] = mapped_column(String)
        canonical_id: Mapped[str] = mapped_column(String)
        current_image_uri: Mapped[Optional[str]] = mapped_column(String, nullable=True)
        last_built_at: Mapped[Optional[datetime]] = mapped_column(DateTime, nullable=True)
        created_at: Mapped[datetime] = mapped_column(DateTime, default=datetime.utcnow)
        is_compliant: Mapped[bool] = mapped_column(Boolean, default=True, server_default="true")
        status: Mapped[str] = mapped_column(String(50), default="DRAFT", server_default="'DRAFT'")
        bom_captured: Mapped[bool] = mapped_column(Boolean, default=False, server_default="false")


    class ImageBOM(EEBase):
        __tablename__ = "image_boms"
        id: Mapped[str] = mapped_column(String(36), primary_key=True)
        template_id: Mapped[str] = mapped_column(String(36))
        # No FK — puppet_templates is in EEBase but CONTEXT.md prohibits FKs to avoid cross-base issues
        raw_data_json: Mapped[str] = mapped_column(Text)
        created_at: Mapped[datetime] = mapped_column(DateTime, default=datetime.utcnow)


    class PackageIndex(EEBase):
        __tablename__ = "package_index"
        id: Mapped[str] = mapped_column(String(36), primary_key=True)
        template_id: Mapped[str] = mapped_column(String(36))
        # No FK — same reason as ImageBOM
        name: Mapped[str] = mapped_column(String(255), index=True)
        version: Mapped[str] = mapped_column(String(50), index=True)
        type: Mapped[str] = mapped_column(String(20))  # 'pip' or 'apt'
    ```
