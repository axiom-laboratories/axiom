    ```python
    import os
    import json
    import time
    import base64
    import logging
    from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PublicKey
    from cryptography.exceptions import InvalidSignature

    logger = logging.getLogger(__name__)

    # 32-byte raw Ed25519 public key for licence verification.
    # Generated with Ed25519PrivateKey.generate(); private key held by Axiom Labs.
    # Replace placeholder with real key before release.
    # This constant is compiled into the .so — no file I/O, fully offline.
    _LICENCE_PUBLIC_KEY_BYTES: bytes = b'\x00' * 32  # placeholder — replace before release


    def _parse_licence(key_str: str):
        """Parse and verify AXIOM_LICENCE_KEY wire format.

        Wire format: base64url(json_payload).base64url(ed25519_sig)
        Returns parsed payload dict on success, None on any failure.

        Order is deliberate: verify signature against raw bytes BEFORE parsing JSON.
        This prevents a tampered payload reaching json.loads().
        """
        try:
            parts = key_str.strip().split('.')
            if len(parts) != 2:
                return None
            payload_bytes = base64.urlsafe_b64decode(parts[0] + '==')
            sig_bytes = base64.urlsafe_b64decode(parts[1] + '==')
            pub = Ed25519PublicKey.from_public_bytes(_LICENCE_PUBLIC_KEY_BYTES)
            pub.verify(sig_bytes, payload_bytes)  # raises InvalidSignature if tampered
            return json.loads(payload_bytes)
        except (InvalidSignature, Exception):
            return None
    ```

    ```python
    # --- Licence check — MUST be first, before any model imports ---

    licence = _parse_licence(licence_key)
    if licence is None:
        logger.warning("Invalid licence signature — EE features disabled")
        return

    exp = licence.get('exp', 0)
    if exp < int(time.time()):
        from datetime import datetime, timezone
        exp_date = datetime.fromtimestamp(exp, tz=timezone.utc).strftime('%Y-%m-%d')
        logger.warning(f"Licence expired on {exp_date} — EE features disabled")

    logger.info(
    ```
