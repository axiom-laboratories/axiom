```python
def incus_exec(container_name: str, cmd: str, timeout: int = 30) -> subprocess.CompletedProcess:
    """Run a bash command inside the named Incus container; return CompletedProcess."""
    return subprocess.run(
        ["incus", "exec", container_name, "--", "bash", "-c", cmd],
        capture_output=True,
        text=True,
        timeout=timeout,
    )

def wait_for_stack_api(container_name: str, timeout: int = API_READINESS_TIMEOUT) -> bool:
    """Poll https://172.17.0.1:8443 inside the LXC container until HTTP 200/301 or timeout."""
    deadline = time.time() + timeout
    attempt = 0
    while time.time() < deadline:
        attempt += 1
        result = incus_exec(
            container_name,
            "curl -k -s -o /dev/null -w '%{http_code}' https://172.17.0.1:8443",
            timeout=15,
        )
        status = result.stdout.strip()
        if status in ("200", "301"):
            print(f"\n✓ Stack ready on {container_name} (attempt {attempt})")
            return True
        print(f".", end="", flush=True)
        time.sleep(5)
    print(f"\n✗ Stack timeout on {container_name} after {timeout}s")
    return False

def wait_for_api_ready(container_name: str, endpoint: str = "/api/features", timeout: int = API_READINESS_TIMEOUT) -> bool:
    """Poll an API endpoint inside the LXC container until HTTP 200 or timeout."""
    deadline = time.time() + timeout
    while time.time() < deadline:
        try:
            result = incus_exec(
                container_name,
                f"curl -k -s -f https://172.17.0.1:8001{endpoint}",
                timeout=10,
            )
            if result.returncode == 0:
                print(f"✓ API ready ({endpoint})")
                return True
        except Exception:
            pass
        print(".", end="", flush=True)
        time.sleep(3)
    print(f"\n✗ API timeout ({endpoint}) after {timeout}s")
    return False
```

```python
@pytest.mark.timeout(120)
def test_ce_table_count(axiom_ce_stack):
    """
    VAL-01: CE-only install creates exactly 15 tables; no EE schema present.
    
    Assertion: SELECT count(*) FROM pg_tables WHERE schemaname='public' == 15
    (excluding apscheduler_jobs, which is APScheduler internal).
    """
    container_name = axiom_ce_stack["lxc_name"]
    
    # Run psql via docker exec inside the LXC
    query = (
        "SELECT count(*) FROM pg_tables "
        "WHERE schemaname='public' AND tablename != 'apscheduler_jobs';"
    )
    
    result = subprocess.run(
        [
            "incus", "exec", container_name, "--", "bash", "-c",
            f"docker exec puppeteer-db-1 psql -U puppet -d puppet_db -t -c \"{query}\""
        ],
        capture_output=True,
        text=True,
        timeout=30,
    )
    
    assert result.returncode == 0, (
        f"psql command failed (exit {result.returncode}): {result.stderr}"
    )
    
    count = int(result.stdout.strip())
    assert count == EXPECTED_CE_TABLE_COUNT, (
        f"Expected {EXPECTED_CE_TABLE_COUNT} CE tables, got {count}. "
        f"EE schema may be present. Output: {result.stdout}"
    )
```

```python
@pytest.mark.timeout(90)
def test_ce_stub_routes_return_402(axiom_ce_stack, get_ce_admin_token):
    """
    VAL-03: All 7 EE stub routes return HTTP 402 (Payment Required) on CE install.
    
    Routes tested:
    - GET /api/blueprints
    - GET /api/puppet-templates
    - GET /api/smelter/ingredients
    - GET /api/vault/config
    - GET /api/siem/config
    - GET /api/executions
    - POST /api/executions/submit
    
    Each must return 402 on CE, confirming EE features are gated.
    """
    base_url = axiom_ce_stack["base_url"]
    token = get_ce_admin_token()
    headers = {"Authorization": f"Bearer {token}"}
    
    results = []
    for method, path in CE_STUB_ROUTES:
        try:
            if method == "GET":
                resp = requests.get(
                    f"{base_url}{path}",
                    headers=headers,
                    verify=False,
                    timeout=10,
                )
            elif method == "POST":
                resp = requests.post(
                    f"{base_url}{path}",
                    headers=headers,
                    json={},  # Empty body for stub test
                    verify=False,
                    timeout=10,
                )
            else:
                resp = None
            
            passed = resp.status_code == 402
            results.append((method, path, resp.status_code, passed))
            
            if not passed:
                print(f"\n[FAIL] {method} {path} returned {resp.status_code}, expected 402")
        except Exception as exc:
            print(f"\n[FAIL] {method} {path} raised {type(exc).__name__}: {exc}")
            results.append((method, path, None, False))
    
    # Assert all routes returned 402
    failed = [r for r in results if not r[3]]
    assert not failed, (
        f"Expected all EE stub routes to return 402, but {len(failed)} failed:\n"
        + "\n".join(f"  {m} {p} -> {s}" for m, p, s, _ in failed)
    )
    
    print(f"✓ VAL-03 PASS: All {len(CE_STUB_ROUTES)} EE stub routes return 402 on CE")
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Test Code → API Server | Test harnesses interact with live puppeteer API inside untrusted LXC containers; self-signed TLS certs; JWT auth required |
| Test Code → Database | psql queries run via docker exec inside LXC; output parsed as integers/strings |
| subprocess.run() calls | All incus_exec and docker exec calls receive user input (paths, env vars); shell injection risk if not escaped |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-173-01 | Injection | incus_exec() bash -c | mitigate | All commands are hardcoded in plan; no user input interpolated into shell commands without validation |
| T-173-02 | Spoofing | admin token via curl | mitigate | Token obtained via HTTPS (with -k for self-signed); verified by checking response JSON structure before use |
| T-173-03 | Tampering | PostgreSQL query results | accept | Query output is parsed as integers; invalid integers will raise ValueError and fail the test (safe rejection) |
| T-173-04 | Repudiation | Test log output | accept | pytest captures all output; no audit trail required for test-only code |
| T-173-05 | Info Disclosure | ADMIN_PASSWORD in logs | mitigate | Token is obtained once in fixture and reused; password is grepped from env but never logged |
| T-173-06 | DoS | subprocess timeout | mitigate | All subprocess.run() calls include explicit timeout (10–120s); deadline-based polling in fixtures has explicit timeouts |

</threat_model>
