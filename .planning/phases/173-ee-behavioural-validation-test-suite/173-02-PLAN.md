**Helper: inject_licence_and_restart** (function, not a fixture):
```python
def inject_licence_and_restart(
    container_name: str,
    licence_key: str,
    admin_password: str,
    timeout: int = 120,
) -> str:
    """
    Inject AXIOM_LICENCE_KEY into /workspace/.env, restart only the agent container,
    wait for API readiness, and return the fresh admin token.
    
    Args:
        container_name: Incus LXC container name (e.g., "axiom-ee-tests")
        licence_key: Full licence key string (or empty to remove)
        admin_password: Admin password (from axiom_ee_stack fixture)
        timeout: Timeout in seconds for API readiness
    
    Returns:
        Fresh admin JWT token
    """
    # Update .env file inside LXC: replace or add AXIOM_LICENCE_KEY
    if licence_key:
        # Remove old line if present, then append new
        update_cmd = (
            f"sed -i '/^AXIOM_LICENCE_KEY=/d' /workspace/.env && "
            f"echo 'AXIOM_LICENCE_KEY={licence_key}' >> /workspace/.env"
        )
    else:
        # Remove the line entirely (test absence scenario)
        update_cmd = "sed -i '/^AXIOM_LICENCE_KEY=/d' /workspace/.env"
    
    result = incus_exec(container_name, update_cmd, timeout=15)
    if result.returncode != 0:
        raise RuntimeError(f"Failed to update .env: {result.stderr}")
    
    # Restart only the agent container (not full stack)
    restart_cmd = (
        "cd /workspace && "
        "docker compose --env-file .env -f compose.cold-start.yaml restart agent"
    )
    result = incus_exec(container_name, restart_cmd, timeout=60)
    if result.returncode != 0:
        raise RuntimeError(f"Failed to restart agent: {result.stderr}")
    
    # Wait for API to be ready
    ready = wait_for_api_ready(container_name, endpoint="/api/licence", timeout=timeout)
    if not ready:
        raise RuntimeError(f"API not ready on {container_name} after licence injection")
    
    # Retrieve fresh admin token using the injected/changed licence key
    for attempt in range(1, 6):
        result = incus_exec(
            container_name,
            (
                f"curl -k -s -X POST https://172.17.0.1:8001/auth/login "
                f"-d 'username=admin&password={admin_password}' "
                f"| python3 -c \"import json,sys; "
                f"d=json.load(sys.stdin); print(d.get('access_token', ''))\" || echo ''"
            ),
            timeout=15,
        )
        token = result.stdout.strip()
        if token:
            return token
        time.sleep(3)
    
    raise RuntimeError("Failed to obtain admin token after licence injection")
```
