class TestParallelGate:
    async def test_parallel_fan_out(self, session, workflow_factory, job_factory):
        """PARALLEL gate marks COMPLETED immediately; fans out to downstream."""
        # Create workflow: start -> parallel (splits to A and B) -> join
        workflow = await workflow_factory(
            name="parallel_fan_out",
            steps=[
                {"id": "start", "node_type": "SCRIPT", "scheduled_job_id": "...", "config_json": None},
                {"id": "parallel", "node_type": "PARALLEL", "scheduled_job_id": None, "config_json": None},
                {"id": "branch_a", "node_type": "SCRIPT", "scheduled_job_id": "...", "config_json": None},
                {"id": "branch_b", "node_type": "SCRIPT", "scheduled_job_id": "...", "config_json": None}
            ],
            edges=[
                {"from_step_id": "start", "to_step_id": "parallel", "branch_name": None},
                {"from_step_id": "parallel", "to_step_id": "branch_a", "branch_name": None},
                {"from_step_id": "parallel", "to_step_id": "branch_b", "branch_name": None}
