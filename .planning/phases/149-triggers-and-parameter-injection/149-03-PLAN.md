```python
@router.post("/api/workflows/{workflow_id}/webhooks", response_model=WorkflowWebhookResponse, status_code=201)
async def create_workflow_webhook(
    workflow_id: str,
    webhook_create: WorkflowWebhookCreate,
    current_user: User = Depends(require_permission("workflows:write")),
    db: AsyncSession = Depends(get_db)
) -> WorkflowWebhookResponse:
    """Create a webhook endpoint for a Workflow. Returns plaintext secret once at creation; never again."""
    # Verify workflow exists and user owns it (or is admin)
    workflow = await db.get(Workflow, workflow_id)
    if workflow is None:
        raise HTTPException(status_code=404, detail="Workflow not found")
    
    # Generate webhook ID and plaintext secret
    webhook_id = str(uuid4())
    plaintext_secret = secrets.token_urlsafe(32)  # ~256 bits of entropy
    
    # Hash secret for storage
    secret_hash = security.hash_webhook_secret(plaintext_secret)
    
    # Create webhook in DB
    webhook = WorkflowWebhook(
        id=webhook_id,
        workflow_id=workflow_id,
        name=webhook_create.name,
        secret_hash=secret_hash,
        created_at=datetime.utcnow()
    )
    db.add(webhook)
    await db.commit()
    
    # Audit log
    await audit("create_webhook", {
        "workflow_id": workflow_id,
        "webhook_id": webhook_id,
        "webhook_name": webhook_create.name
    }, db)
    
    # Return response with plaintext secret (only time it's exposed)
    response = WorkflowWebhookResponse(
        id=webhook_id,
        workflow_id=workflow_id,
        name=webhook_create.name,
        secret=plaintext_secret,  # ONLY in creation response
        created_at=webhook.created_at
    )
    return response
```

Key points:
- Gated on `workflows:write` permission
- Generate webhook_id via uuid4()
- Generate plaintext secret via secrets.token_urlsafe(32) — ~256 bits of entropy
- Hash secret immediately using security.hash_webhook_secret()
- Store hash only (never store plaintext)
- Return plaintext secret ONLY in creation response (201)
- Subsequent GET requests return secret=None
- Audit log creation

```python
@router.get("/api/workflows/{workflow_id}/webhooks", response_model=List[WorkflowWebhookResponse])
async def list_workflow_webhooks(
    workflow_id: str,
    current_user: User = Depends(require_permission("workflows:write")),
    db: AsyncSession = Depends(get_db)
) -> List[WorkflowWebhookResponse]:
    """List all webhooks for a Workflow (secret field omitted for security)."""
    # Verify workflow exists
    workflow = await db.get(Workflow, workflow_id)
    if workflow is None:
        raise HTTPException(status_code=404, detail="Workflow not found")
    
    # Fetch all webhooks for this workflow
    stmt = select(WorkflowWebhook).where(WorkflowWebhook.workflow_id == workflow_id)
    result = await db.execute(stmt)
    webhooks = result.scalars().all()
    
    # Map to response (secret=None for all list responses)
    return [
        WorkflowWebhookResponse(
            id=w.id,
            workflow_id=w.workflow_id,
            name=w.name,
            secret=None,  # Never expose secret in list
            created_at=w.created_at
        )
        for w in webhooks
    ]

@router.delete("/api/workflows/{workflow_id}/webhooks/{webhook_id}", status_code=204)
async def delete_workflow_webhook(
    workflow_id: str,
    webhook_id: str,
    current_user: User = Depends(require_permission("workflows:write")),
    db: AsyncSession = Depends(get_db)
) -> None:
    """Revoke a webhook endpoint. Future trigger requests will fail with 404."""
    # Verify webhook exists and belongs to this workflow
    webhook = await db.get(WorkflowWebhook, webhook_id)
    if webhook is None or webhook.workflow_id != workflow_id:
        raise HTTPException(status_code=404, detail="Webhook not found")
    
    # Delete webhook
    await db.delete(webhook)
    await db.commit()
    
    # Audit log
    await audit("delete_webhook", {
        "workflow_id": workflow_id,
        "webhook_id": webhook_id
    }, db)
```

- Both gated on workflows:write
- Verify workflow_id matches webhook.workflow_id to prevent leaking webhooks across workflows

```python
@router.post("/api/webhooks/{webhook_id}/trigger", status_code=202)
async def trigger_webhook(
    webhook_id: str,
    request: Request,  # FastAPI Request object for raw body + headers
    db: AsyncSession = Depends(get_db)
) -> Dict[str, str]:
    """Unauthenticated webhook trigger endpoint. Validates HMAC signature before running workflow."""
    
    # Read raw request body (required for HMAC verification)
    try:
        body_bytes = await request.body()
    except Exception as e:
        logger.error(f"Error reading webhook body: {e}")
        raise HTTPException(status_code=400, detail="Invalid request body")
    
    # Extract signature header
    signature_header = request.headers.get("X-Hub-Signature-256", "")
    if not signature_header:
        logger.warning(f"Webhook {webhook_id} missing X-Hub-Signature-256 header")
        raise HTTPException(status_code=401, detail="Missing signature header")
    
    # Fetch webhook
    webhook = await db.get(WorkflowWebhook, webhook_id)
    if webhook is None:
        # Log but don't expose webhook existence
        logger.warning(f"Webhook trigger attempt for unknown webhook {webhook_id}")
        raise HTTPException(status_code=404, detail="Webhook not found")
    
    # Retrieve plaintext secret from DB (we only store hash)
    # DECISION: Since we store bcrypt hash, we need a way to get plaintext.
    # Two approaches per RESEARCH.md:
    # A) Store plaintext in separate encrypted field (Fernet) alongside hash
    # B) Store plaintext temporarily in memory at creation time (not ideal)
    # For Phase 149, assume webhook.secret_plaintext field exists (or refactor to Fernet-encrypted field)
    # If only hash exists, webhook trigger cannot work — this is a known limitation.
    # Planner will choose: either add secret_plaintext field to WorkflowWebhook, or use alternative hashing (SHA-256 instead of bcrypt).
    #
    # For now, assume secret_plaintext field exists:
    plaintext_secret = webhook.secret_plaintext  # This field must be added in Plan 01 refactor OR use alternative design
    
    # Verify signature
    if not security.verify_webhook_signature(signature_header, body_bytes, plaintext_secret):
        logger.warning(f"Webhook {webhook_id} signature verification failed")
        raise HTTPException(status_code=401, detail="Signature verification failed")
    
    # Parse request body as parameters dict
    try:
        parameters = await request.json()  # Should be JSON object
        if not isinstance(parameters, dict):
            parameters = {}
    except Exception as e:
        logger.warning(f"Webhook {webhook_id} invalid JSON body: {e}")
        parameters = {}  # Treat as empty if JSON invalid
    
    # Trigger workflow run
    try:
        run = await workflow_service.start_run(
            workflow_id=webhook.workflow_id,
            parameters=parameters,
            trigger_type="WEBHOOK",
            triggered_by=webhook.name,  # Webhook name in audit trail
            db=db
        )
        
        # Audit log
        await audit("trigger_webhook", {
            "webhook_id": webhook_id,
            "workflow_id": webhook.workflow_id,
            "run_id": run.id
        }, db)
        
        return {"run_id": run.id}  # 202 + run_id
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Error triggering webhook {webhook_id}: {e}")
        raise HTTPException(status_code=500, detail="Failed to trigger workflow")
```

Key points:
- NO JWT authentication required; HMAC signature is the auth mechanism
- Extract signature from `X-Hub-Signature-256` header (format: "sha256=<hex>")
- Read raw request body for HMAC verification
- Verify signature using security.verify_webhook_signature()
- 401 if signature mismatch or header missing
- 404 if webhook_id unknown
- Parameters: POST body JSON dict; merged with workflow defaults in start_run()
- trigger_type="WEBHOOK", triggered_by=webhook.name (for audit trail)
- Return 202 (Accepted) + {"run_id": run.id}

CRITICAL DECISION: Webhook secret storage strategy
- Current design stores bcrypt hash only, but bcrypt is one-way (can't extract plaintext)
- To verify HMAC, endpoint needs plaintext secret
- RESOLUTION (per CONTEXT.md discretion): Planner chooses between:
  Option A: Add secret_plaintext field to WorkflowWebhook, stored encrypted via Fernet (alongside bcrypt hash)
  Option B: Use SHA-256 hash instead of bcrypt (unsalted, less secure but simpler)
  Option C: Store plaintext in memory during creation, re-hash on each trigger (not feasible)

For now, assume Option A (secret_plaintext Fernet-encrypted field added to ORM in Plan 01 refactor).

```python
@router.patch("/api/workflows/{workflow_id}", response_model=WorkflowResponse)
async def update_workflow(
    workflow_id: str,
    workflow_update: WorkflowUpdate,
    current_user: User = Depends(require_permission("workflows:write")),
    db: AsyncSession = Depends(get_db)
) -> WorkflowResponse:
    """Update a Workflow definition. Validates DAG and parameter constraints."""
    
    # Fetch workflow
    workflow = await db.get(Workflow, workflow_id)
    if workflow is None:
        raise HTTPException(status_code=404, detail="Workflow not found")
    
    # Update fields
    if workflow_update.name is not None:
        workflow.name = workflow_update.name
    if workflow_update.definition is not None:
        workflow.definition = workflow_update.definition
        # Validate DAG (existing logic from Phase 146)
        await workflow_service.validate_workflow_definition(workflow)
    if workflow_update.schedule_cron is not None:
        # NEW Phase 149: Validate cron against parameters
        await db.refresh(workflow, ["parameters"])
        for param in workflow.parameters:
            if param.default_value is None:
                raise HTTPException(
                    status_code=422,
                    detail=f"Cannot schedule: required parameter '{param.name}' has no default"
                )
        workflow.schedule_cron = workflow_update.schedule_cron
        # Sync cron with APScheduler (existing scheduler_service.sync_workflow_crons call)
        await scheduler_service.sync_workflow_crons()
    if workflow_update.is_paused is not None:
        workflow.is_paused = workflow_update.is_paused
        # If pausing, sync crons to remove APScheduler job
        if workflow.is_paused and workflow.schedule_cron is not None:
            await scheduler_service.sync_workflow_crons()
    
    workflow.updated_at = datetime.utcnow()
    await db.commit()
    
    # Audit log
    await audit("update_workflow", {
        "workflow_id": workflow_id,
        "schedule_cron": workflow_update.schedule_cron,
        "is_paused": workflow_update.is_paused
    }, db)
    
    return await workflow_service._workflow_to_response(db, workflow)
```
