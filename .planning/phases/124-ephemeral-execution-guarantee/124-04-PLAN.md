import pytest
from httpx import AsyncClient
from agent_service.main import app
import os


@pytest.mark.asyncio
class TestComposeValidation:
    """Test compose generator validation of execution_mode."""

    async def test_compose_rejects_direct_mode(self, client: AsyncClient, valid_token: str):
        """Compose generator rejects EXECUTION_MODE=direct."""
        response = await client.get(
            "/api/node/compose",
            params={"execution_mode": "direct", "token": valid_token}
        )
        assert response.status_code == 400
        assert "not supported" in response.json()["detail"]

    async def test_compose_accepts_docker_mode(self, client: AsyncClient, valid_token: str):
        """Compose generator accepts EXECUTION_MODE=docker."""
        response = await client.get(
            "/api/node/compose",
            params={"execution_mode": "docker", "token": valid_token}
        )
        assert response.status_code == 200
        assert "EXECUTION_MODE" in response.text

    async def test_compose_accepts_podman_mode(self, client: AsyncClient, valid_token: str):
        """Compose generator accepts EXECUTION_MODE=podman."""
        response = await client.get(
            "/api/node/compose",
            params={"execution_mode": "podman", "token": valid_token}
        )
        assert response.status_code == 200
        assert "EXECUTION_MODE" in response.text

    async def test_compose_accepts_auto_mode(self, client: AsyncClient, valid_token: str):
        """Compose generator accepts EXECUTION_MODE=auto (default)."""
        response = await client.get(
            "/api/node/compose",
            params={"token": valid_token}
            # No execution_mode param — should default to "auto"
        )
        assert response.status_code == 200
        assert "EXECUTION_MODE" in response.text

    async def test_compose_error_message_helpful(self, client: AsyncClient, valid_token: str):
        """Error message for direct mode is actionable."""
        response = await client.get(
            "/api/node/compose",
            params={"execution_mode": "direct", "token": valid_token}
        )
        detail = response.json()["detail"]
        assert "docker" in detail.lower() or "podman" in detail.lower()
        # Message should suggest alternatives
        assert any(x in detail.lower() for x in ["docker", "podman", "auto"])
```

**Why:** Validates that compose endpoint properly rejects direct mode and accepts valid modes. Error message is verified to be helpful.

import pytest
from agent_service.models import HeartbeatPayload
from agent_service.services import job_service
from agent_service.db import Node


@pytest.mark.asyncio
class TestHeartbeatHandlerExecutionMode:
    """Test job_service.handle_heartbeat() updates node.execution_mode."""

    async def test_handle_heartbeat_persists_execution_mode(self, db_session, sample_node):
        """handle_heartbeat() updates node.execution_mode from payload."""
        # Create sample heartbeat with execution_mode
        hb = HeartbeatPayload(
            node_id=sample_node.node_id,
            hostname="testhost",
            execution_mode="docker"
        )

        # Call handler
        await job_service.handle_heartbeat(hb, db_session)

        # Verify DB updated
        assert sample_node.execution_mode == "docker"

    async def test_handle_heartbeat_with_podman_mode(self, db_session, sample_node):
        """handle_heartbeat() persists podman mode."""
        hb = HeartbeatPayload(
            node_id=sample_node.node_id,
            hostname="testhost",
            execution_mode="podman"
        )
        await job_service.handle_heartbeat(hb, db_session)
        assert sample_node.execution_mode == "podman"

    async def test_handle_heartbeat_execution_mode_none(self, db_session, sample_node):
        """handle_heartbeat() handles missing execution_mode (old nodes)."""
        hb = HeartbeatPayload(
            node_id=sample_node.node_id,
            hostname="testhost"
            # No execution_mode — old node
        )
        await job_service.handle_heartbeat(hb, db_session)
        # Should not crash; execution_mode may be None
        assert sample_node.execution_mode is None or isinstance(sample_node.execution_mode, str)

    async def test_handle_heartbeat_updates_alongside_cgroup(self, db_session, sample_node):
        """handle_heartbeat() updates execution_mode alongside cgroup fields."""
        hb = HeartbeatPayload(
            node_id=sample_node.node_id,
            hostname="testhost",
            execution_mode="docker",
            detected_cgroup_version="v2"
        )
        await job_service.handle_heartbeat(hb, db_session)
        assert sample_node.execution_mode == "docker"
        assert sample_node.detected_cgroup_version == "v2"
```

**Why:** Validates that the heartbeat handler correctly persists execution_mode to the database, along with other heartbeat fields like cgroup detection.
