```python
def get_container_id(service_name):
    """Get container ID for a service by querying docker ps."""
    result = subprocess.run(
        ['docker', 'ps', '--filter', f'name={service_name}', '-q'],
        capture_output=True, text=True, check=True, timeout=5
    )
    container_id = result.stdout.strip()
    if not container_id:
        raise RuntimeError(f"Container '{service_name}' not running")
    return container_id
```

**Alternative approach (more robust):**
Use multi-line matching to handle lines split with backslash:
```python
def test_podman_package_not_installed():
    """CONT-07: Verify 'podman' package not installed in node image."""
    with open('puppets/Containerfile.node', 'r') as f:
        content = f.read()
    
    # Remove line continuations (\ at end of line) and normalize
    content = re.sub(r'\\\n\s*', ' ', content)
    
    # Find all apt-get install commands (handles multi-line)
    install_commands = re.findall(r'apt-get install[^;]*', content, re.DOTALL)
    
    for cmd in install_commands:
        assert 'podman' not in cmd, \
            f"podman package in install: {cmd[:100]}..."
```
